185  /// <summary>
186  /// Handles the single parcel server connections
187  /// </summary>
188  /// <param name="connection">The connection</param>
189  /// <returns>The async task</returns>
190  // ReSharper disable once UnusedMethodReturnValue.Local
191  private async Task HandleConnection(TcpClient connection)
192  {
193  try
194  {
195  using (var stream = connection.GetStream())
196  {
197  var guidBytes = new byte[16];
198  var readBytes = await this.ReadWithTimeout(stream, guidBytes, 0, 16, this.readTimeout);
199  if (readBytes != 16)
200  {
201  await stream.WriteAsync(new[] { (byte)EnParcelServerResponseCode.BadRequest }, 0, 1);
202  return;
203  }
204 
205  Guid uid;
206  try
207  {
208  uid = new Guid(guidBytes);
209  }
210  catch
211  {
212  await stream.WriteAsync(new[] { (byte)EnParcelServerResponseCode.BadRequest }, 0, 1);
213  return;
214  }
215 
216  Parcel parcel;
217  if (!Parcels.TryRemove(uid, out parcel))
218  {
219  await stream.WriteAsync(new[] { (byte)EnParcelServerResponseCode.NotFound }, 0, 1);
220  return;
221  }
222 
223  await stream.WriteAsync(new[] { (byte)EnParcelServerResponseCode.Ok }, 0, 1);
224  var data = parcel.Payload.SerializeToAkka(this.sys);
225  await stream.WriteAsync(BitConverter.GetBytes(data.Length), 0, 4);
226  await stream.WriteAsync(data, 0, data.Length);
227  }
228  }
229  catch (Exception e)
230  {
231  Log.Logger.Error(e, "{Type}: error while handling connection", this.GetType().Name);
232  }
233  finally
234  {
235 #if APPDOMAIN
236  connection.Close();
237 #elif CORECLR
238  connection.Dispose();
239 #else
240 #error Not implemented
241  throw new InvalidOperationException();
242 #endif
243  }
244  }

283  /// <summary>
284  /// Reads from the stream with provided timeout
285  /// </summary>
286  /// <param name="stream">The stream to read from</param>
287  /// <param name="buffer">The byte buffer</param>
288  /// <param name="offset">The stream offset</param>
289  /// <param name="length">The number of bytes to read</param>
290  /// <param name="timeout">The timeout</param>
291  /// <returns>The number of read bytes</returns>
292  private async Task<int> ReadWithTimeout(
293  NetworkStream stream,
294  byte[] buffer,
295  int offset,
296  int length,
297  TimeSpan timeout)
298  {
299  var tokenSource = new CancellationTokenSource();
300  var ct = tokenSource.Token;
301 
302  var task = stream.ReadAsync(buffer, offset, length, ct);
303 
304  if (await Task.WhenAny(task, Task.Delay(timeout, ct)) == task)
305  {
306  tokenSource.Cancel();
307  return task.Result;
308  }
309 
310  tokenSource.Cancel();
311  throw new TimeoutException();
312  }

314  /// <summary>
315  /// Sends parcel notification
316  /// </summary>
317  /// <param name="parcel">The parcel</param>
318  private void SendNotification(Parcel parcel)
319  {
320  var notification = new ParcelNotification
321  {
322  Host = this.host,
323  Port = this.port,
324  Uid = parcel.Uid,
325  PayloadTypeName =
326  parcel.Payload.GetType().AssemblyQualifiedName
327  };
328 
329  object envelope = null;
330  foreach (var notificationEnveloper in this.envelopers)
331  {
332  envelope = notificationEnveloper.Envelope(parcel, notification);
333  if (envelope != null)
334  {
335  break;
336  }
337  }
338 
339  parcel.Recipient.Tell(envelope ?? notification, this.Sender);
340  }
