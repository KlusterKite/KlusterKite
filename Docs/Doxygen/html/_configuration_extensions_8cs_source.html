121  public static IEnumerable<CompatibleTemplate> GetCompatibleTemplates(
122  this Configuration configuration,
123  ConfigurationContext context)
124  {
125  if (configuration?.Settings?.NodeTemplates == null)
126  {
127  yield break;
128  }
129 
130  var currentConfiguration =
131  context.Configurations.Include(nameof(Configuration.CompatibleTemplatesBackward))
132  .FirstOrDefault(r => r.State == EnConfigurationState.Active);
133 
134  if (currentConfiguration?.Settings?.NodeTemplates == null)
135  {
136  yield break;
137  }
138 
139  foreach (var template in currentConfiguration.Settings.NodeTemplates)
140  {
141  var currentTemplate = configuration.Settings.NodeTemplates.FirstOrDefault(t => t.Code == template.Code);
142  if (currentTemplate == null)
143  {
144  continue;
145  }
146 
147  if (currentTemplate.Configuration != template.Configuration)
148  {
149  continue;
150  }
151 
152  var oldRequirements = string.Join(
153  "; ",
154  template.PackageRequirements.OrderBy(p => p.Id).Select(p => $"{p.Id} {p.SpecificVersion}"));
155  var newRequirements = string.Join(
156  "; ",
157  template.PackageRequirements.OrderBy(p => p.Id).Select(p => $"{p.Id} {p.SpecificVersion}"));
158 
159  if (oldRequirements != newRequirements)
160  {
161  continue;
162  }
163 
164  var needPackageUpdate = false;
165  foreach (var requirement in currentTemplate.PackageRequirements.Where(r => r.SpecificVersion == null))
166  {
167  var oldVersion = configuration.Settings.Packages.FirstOrDefault(p => p.Id == requirement.Id);
168  var newVersion = currentConfiguration.Settings.Packages.FirstOrDefault(p => p.Id == requirement.Id);
169  if (newVersion == null || oldVersion?.Version != newVersion.Version)
170  {
171  needPackageUpdate = true;
172  break;
173  }
174  }
175 
176  if (needPackageUpdate)
177  {
178  continue;
179  }
180 
181  yield return
182  new CompatibleTemplate
183  {
184  CompatibleConfigurationId = currentConfiguration.Id,
185  ConfigurationId = configuration.Id,
186  TemplateCode = template.Code
187  };
188 
189  foreach (
190  var compatible in currentConfiguration.CompatibleTemplatesBackward.Where(ct => ct.TemplateCode == template.Code))
191  {
192  yield return
193  new CompatibleTemplate
194  {
195  CompatibleConfigurationId = compatible.CompatibleConfigurationId,
196  ConfigurationId = configuration.Id,
197  TemplateCode = template.Code
198  };
199  }
200  }

263  private static async Task<Tuple<Dictionary<string, IPackageSearchMetadata>, List<ErrorDescription>>>
264  CheckPackages(Configuration configuration, List<string> supportedFrameworks, IPackageRepository nugetRepository)
265  {
266  var definedPackages = new Dictionary<string, IPackageSearchMetadata>();
267  var errors = new List<ErrorDescription>();
268  var packages = await Task.WhenAll(
269  configuration.Settings.Packages
270  .Select(d => new { Description = d, Version = NuGetVersion.TryParse(d.Version, out var v) ? v : null })
271  .Select(
272  async p => new
273  {
274  p.Description, p.Version,
275  Package = p.Version != null
276  ? await nugetRepository.GetAsync(p.Description.Id, p.Version)
277  : null
278  }));
279 
280  foreach (var package in packages)
281  {
282  if (package.Version == null)
283  {
284  errors.Add(
285  new ErrorDescription(
286  $"configuration.packages[\"{package.Description.Id}\"]",
287  "Package version could not be parsed"));
288  }
289  else if (package.Package == null)
290  {
291  errors.Add(
292  new ErrorDescription(
293  $"configuration.packages[\"{package.Description.Id}\"]",
294  "Package of specified version could not be found in the nuget repository"));
295  }
296  else
297  {
298  definedPackages[package.Description.Id] = package.Package;
299  }
300  }
301 
302  var frameworks = supportedFrameworks.Select(
303  f => NuGetFramework.ParseFrameworkName(f, new DefaultFrameworkNameProvider()))
304  .ToList();
305 
306  errors.AddRange(definedPackages.Values
307  .SelectMany(
308  p => p.DependencySets.Where(
309  s => frameworks.Any(
310  f => NuGetFrameworkUtility.IsCompatibleWithFallbackCheck(f, s.TargetFramework))))
311  .SelectMany(s => s.Packages).GroupBy(p => p.Id).Select(
312  r =>
313  {
314  var packageField = $"configuration.packages[\"{r.Key}\"]";
315  IPackageSearchMetadata definedPackage;
316  if (!definedPackages.TryGetValue(r.Key, out definedPackage))
317  {
318  return new ErrorDescription(
319  packageField,
320  "Package is not defined");
321  }
322 
323  if (r.Any(rc => !rc.VersionRange.Satisfies(definedPackage.Identity.Version)))
324  {
325  return new ErrorDescription(
326  packageField,
327  "Package doesn't satisfy other packages version requirements");
328  }
329 
330  return null;
331  })
332  .Where(e => e != null));
333  return Tuple.Create(definedPackages, errors);
334  }

354  private static async Task<List<ErrorDescription>> CheckTemplatesPackages(
355  [NotNull] Configuration configuration,
356  [NotNull] List<string> supportedFrameworks,
357  [NotNull] Dictionary<string, IPackageSearchMetadata> definedPackages,
358  [NotNull] IPackageRepository nugetRepository)
359  {
360  if (configuration == null)
361  {
362  throw new ArgumentNullException(nameof(configuration));
363  }
364 
365  if (configuration.Settings == null)
366  {
367  throw new ArgumentNullException(nameof(configuration.Settings));
368  }
369 
370  if (configuration.Settings.NodeTemplates == null)
371  {
372  throw new ArgumentNullException(nameof(configuration.Settings.NodeTemplates));
373  }
374 
375  if (supportedFrameworks == null)
376  {
377  throw new ArgumentNullException(nameof(supportedFrameworks));
378  }
379 
380  if (definedPackages == null)
381  {
382  throw new ArgumentNullException(nameof(definedPackages));
383  }
384 
385  if (nugetRepository == null)
386  {
387  throw new ArgumentNullException(nameof(nugetRepository));
388  }
389 
390  var errors = new List<ErrorDescription>();
391 
392  foreach (var template in configuration.Settings.GetAllTemplates())
393  {
394  if (template == null)
395  {
396  continue;
397  }
398 
399  var templateField = template is NodeTemplate
400  ? $"configuration.nodeTemplates[\"{template.Code}\"]"
401  : $"configuration.migratorTemplates[\"{template.Code}\"]";
402 
403  if (template.PackageRequirements == null || template.PackageRequirements.Count == 0)
404  {
405  errors.Add(new ErrorDescription(templateField, "Package requirements are not set"));
406  continue;
407  }
408 
409  if (template is MigratorTemplate)
410  {
411  // Every MigratorTemplate should contain this package for execution
412  if (template.PackageRequirements.All(p => p.Id != "KlusterKite.NodeManager.Migrator.Executor"))
413  {
414  template.PackageRequirements.Add(new NodeTemplate.PackageRequirement("KlusterKite.NodeManager.Migrator.Executor", null));
415  }
416  }
417 
418  var(templatePackages, templateErrors) =
419  await GetTemplateDirectPackages(definedPackages, nugetRepository, template, templateField);
420 
421  errors.AddRange(templateErrors);
422 
423  template.PackagesToInstall = new Dictionary<string, List<PackageDescription>>();
424  foreach (var supportedFramework in supportedFrameworks.Select(
425  f => NuGetFramework.ParseFrameworkName(f, new DefaultFrameworkNameProvider())))
426  {
427  var packagesToInstall = new Dictionary<string, IPackageSearchMetadata>();
428  foreach (var package in templatePackages)
429  {
430  packagesToInstall.Add(package.Key, package.Value);
431  }
432 
433  var queue = new Queue<IPackageSearchMetadata>(templatePackages.Values);
434 
435  while (queue.Count > 0)
436  {
437  var package = queue.Dequeue();
438  if (!packagesToInstall.ContainsKey(package.Identity.Id))
439  {
440  packagesToInstall.Add(package.Identity.Id, package);
441  }
442 
443  var requirementField = $"{templateField}.packageRequirements[\"{package.Identity.Id}\"]";
444  var dependencySet =
445  NuGetFrameworkUtility.GetNearest(package.DependencySets, supportedFramework);
446  if (dependencySet == null || !NuGetFrameworkUtility.IsCompatibleWithFallbackCheck(
447  supportedFramework,
448  dependencySet.TargetFramework))
449  {
450  continue;
451  }
452 
453  foreach (var dependency in dependencySet.Packages)
454  {
455  IPackageSearchMetadata packageToInstall;
456  if (!packagesToInstall.TryGetValue(dependency.Id, out packageToInstall))
457  {
458  if (!definedPackages.TryGetValue(dependency.Id, out packageToInstall))
459  {
460  errors.Add(
461  new ErrorDescription(
462  requirementField,
463  $"Package dependency for {supportedFramework.DotNetFrameworkName} {dependency.Id} {dependency.VersionRange} is missing"));
464  continue;
465  }
466 
467  if (queue.All(p => p.Identity.Id != packageToInstall.Identity.Id))
468  {
469  queue.Enqueue(packageToInstall);
470  }
471  }
472 
473  if (!dependency.VersionRange.Satisfies(packageToInstall.Identity.Version))
474  {
475  errors.Add(
476  new ErrorDescription(
477  requirementField,
478  $"Package dependency for {supportedFramework} {packageToInstall.Identity} doesn't satisfy version requirements {dependency.VersionRange}."));
479  }
480  }
481  }
482 
483  template.PackagesToInstall[supportedFramework.DotNetFrameworkName] = packagesToInstall.Values
484  .Select(
485  p => new PackageDescription { Id = p.Identity.Id, Version = p.Identity.Version.ToString() })
486  .ToList();
487  }
488  }
489 
490  return errors;
491  }

511  private static async Task<Tuple<Dictionary<string, IPackageSearchMetadata>, List<ErrorDescription>>> GetTemplateDirectPackages(
512  Dictionary<string, IPackageSearchMetadata> definedPackages,
513  IPackageRepository nugetRepository,
514  ITemplate nodeTemplate,
515  string templateField)
516  {
517  var errors = new List<ErrorDescription>();
518  var directPackagesToFill = new Dictionary<string, IPackageSearchMetadata>();
519 
520  foreach (var requirement in nodeTemplate.PackageRequirements.Where(r => r.SpecificVersion == null))
521  {
522  IPackageSearchMetadata package;
523  if (!definedPackages.TryGetValue(requirement.Id, out package))
524  {
525  var requirementField = $"{templateField}.packageRequirements[\"{requirement.Id}\"]";
526  errors.Add(
527  new ErrorDescription(
528  requirementField,
529  "Package requirement is not defined in configuration packages"));
530  }
531  else
532  {
533  directPackagesToFill[package.Identity.Id] = package;
534  }
535  }
536 
537  var searchResults = await Task.WhenAll(nodeTemplate.PackageRequirements.Where(r => r.SpecificVersion != null)
538  .Select(r => new { Requirement = r, Version = NuGetVersion.TryParse(r.SpecificVersion, out var v) ? v : null })
539  .Select(
540  async r => new
541  {
542  r.Requirement,
543  r.Version,
544  Package = r.Version != null ? await nugetRepository.GetAsync(
545  r.Requirement.Id,
546  r.Version) : null
547  }));
548 
549  foreach (var result in searchResults)
550  {
551  var requirementField = $"{templateField}.packageRequirements[\"{result.Requirement.Id}\"]";
552  if (result.Version == null)
553  {
554  errors.Add(
555  new ErrorDescription(requirementField, "Package version could not be parsed"));
556  }
557  else if (result.Package == null)
558  {
559  errors.Add(
560  new ErrorDescription(requirementField, "Package could not be found in nuget repository"));
561  }
562  else
563  {
564  directPackagesToFill[result.Requirement.Id] = result.Package;
565  }
566  }
567 
568  return Tuple.Create(directPackagesToFill, errors);
569  }
570  }
