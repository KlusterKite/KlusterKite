53  /// <summary>
54  /// Installs specified packages to specified place
55  /// </summary>
56  /// <param name="packages">
57  /// The list of packages to install
58  /// </param>
59  /// <param name="runtime">
60  /// The current runtime
61  /// </param>

63  /// The target framework to install

72  /// <returns>

74  /// </returns>
75  public static async Task<Dictionary<PackageIdentity, IEnumerable<string>>> Install(
76  this IEnumerable<PackageIdentity> packages,
77  string runtime,
78  NuGetFramework framework,
79  string nugetUrl,
80  string directoryToInstall,
81  Action<string> logAction = null)
82  {
83  var tempDir = Path.Combine(Path.GetTempPath(), Guid.NewGuid().ToString("N"));
84  Directory.CreateDirectory(tempDir);
85 
86  var source = new PackageSource(nugetUrl);
87  var sourceRepository = Repository.Factory.GetCoreV3(source.Source);
88  var sourceCacheContext = new SourceCacheContext();
89 
90  var downloadResource = await sourceRepository.GetResourceAsync<DownloadResource>().ConfigureAwait(false);
91  var packageDownloadContext = new PackageDownloadContext(sourceCacheContext);
92  var result = new Dictionary<PackageIdentity, IEnumerable<string>>();
93  try
94  {
95  var downloadTasks = packages.Select(
96  package => downloadResource.GetDownloadResourceResultAsync(
97  package,
98  packageDownloadContext,
99  tempDir,
100  NullLogger.Instance,
101  CancellationToken.None).ContinueWith(
102  t =>
103  {
104  logAction?.Invoke(
105  t.Result.PackageReader == null
106  ? $"{package.Id} {package.Version} failed to download: {t.Result.Status}"
107  : $"{package.Id} downloaded");
108  return t.Result;
109  }));
110 
111  var downloads = await Task.WhenAll(downloadTasks);
112 
113  if (downloads.Any(d => d.PackageReader == null))
114  {
115  throw new Exception("Could not install all required packages");
116  }
117 
118  var tools = downloads.Where(d => d.PackageReader.GetToolItems().Any()).ToList();
119  var libraries = downloads.Where(d => !d.PackageReader.GetToolItems().Any()).ToList();
120 
121  foreach (var package in tools.OrderBy(d => d.PackageReader.GetIdentity().Id))
122  {
123  var files = ExtractPackage(package, runtime, framework, directoryToInstall).ToList();
124  logAction?.Invoke($"{package.PackageReader.GetIdentity()} extracted as tool with {files.Count} files");
125  result[package.PackageReader.GetIdentity()] = files;
126  }
127 
128  foreach (var package in libraries.OrderBy(d => d.PackageReader.GetIdentity().Id))
129  {
130  var files = ExtractPackage(package, runtime, framework, directoryToInstall).ToList();
131  logAction?.Invoke($"{package.PackageReader.GetIdentity()} extracted as lib with {files.Count} files");
132  result[package.PackageReader.GetIdentity()] = files;
133  }
134 
135  return result;
136  }
137  finally
138  {
139  try
140  {
141  GC.Collect();
142  Directory.Delete(tempDir, true);
143  }
144  catch
145  {
146  // ignore
147  }
148  }
149  }

151  /// <summary>
152  /// Searches for a specific package
153  /// </summary>
154  /// <param name="nugetUrl">
155  /// The nuget server address
156  /// </param>
157  /// <param name="id">
158  /// The package id
159  /// </param>

161  /// The package version.
162  /// </param>
163  /// <returns>
164  /// The package metadata
165  /// </returns>
166  public static async Task<IPackageSearchMetadata> Search(string nugetUrl, string id, NuGetVersion version)
167  {
168  var source = new PackageSource(nugetUrl);
169  var sourceRepository = Repository.Factory.GetCoreV3(source.Source);
170  var resource = await sourceRepository.GetResourceAsync<PackageSearchResource>().ConfigureAwait(false);
171  var result = new List<IPackageSearchMetadata>();
172 
173  const int PageSize = 1000;
174  var position = 0;
175  while (true)
176  {
177  var searchMetadata = (await resource.SearchAsync(
178  id,
179  new SearchFilter(true, null),
180  position,
181  PageSize,
182  NullLogger.Instance,
183  CancellationToken.None)).ToList();
184  var previousCount = result.Count;
185  result.AddRange(searchMetadata);
186  position += PageSize;
187  if (searchMetadata.Any(s => s.Identity.Id == id))
188  {
189  break;
190  }
191 
192  if (result.Count - previousCount < PageSize)
193  {
194  break;
195  }
196  }
197 
198  var package = result.FirstOrDefault(s => s.Identity.Id == id);
199  if (package == null)
200  {
201  return null;
202  }
203 
204  var versions = (await package.GetVersionsAsync()).ToList();
205  return versions.FirstOrDefault(v => v.Version == version)?.PackageSearchMetadata;
206  }

244  /// <summary>
245  /// Extracts the lib files to execution directory
246  /// </summary>
247  /// <param name="package">
248  /// The package to extract
249  /// </param>
250  /// <param name="runtime">
251  /// The current runtime
252  /// </param>
253  /// <param name="frameworkName">
254  /// The current framework name
255  /// </param>
256  /// <param name="executionDir">
257  /// The execution directory to load packages
258  /// </param>
259  /// <param name="logAction">The log writing action</param>
260  /// <returns>
261  /// The list of extracted files
262  /// </returns>
263  private static IEnumerable<string> ExtractPackage(
264  DownloadResourceResult package,
265  string runtime,
266  NuGetFramework frameworkName,
267  string executionDir,
268  Action<string> logAction = null)
269  {
270  try
271  {
272  var id = package.PackageReader.GetIdentity();
273  var files = NuGetFrameworkUtility.GetNearest(package.PackageReader.GetLibItems(), frameworkName)?.Items.ToList()
274  ?? NuGetFrameworkUtility.GetNearest(package.PackageReader.GetToolItems(), frameworkName)?.Items.ToList();
275 
276  if (files == null || files.Count == 0)
277  {
278  var collection = new ContentItemCollection();
279  collection.Load(package.PackageReader.GetFiles());
280 
281  var conventions = new ManagedCodeConventions(runtimeGraph);
282  var criteria = conventions.Criteria.ForFrameworkAndRuntime(
283  NuGetFramework.ParseFrameworkName(
284  PackageRepositoryExtensions.CurrentRuntime,
285  DefaultFrameworkNameProvider.Instance),
286  runtime);
287 
288  files = collection.FindBestItemGroup(criteria, conventions.Patterns.NativeLibraries)?.Items
289  .Select(i => i.Path).ToList();
290  if (files == null || files.Count == 0)
291  {
292  files = collection.FindBestItemGroup(criteria, conventions.Patterns.RuntimeAssemblies)?.Items
293  .Select(i => i.Path).ToList();
294  }
295 
296  if (files == null || files.Count == 0)
297  {
298  return new string[0];
299  }
300  else
301  {
302  logAction?.Invoke($"{id.Id}: {string.Join(", ", files)}");
303  }
304  }
305 
306  foreach (var file in files)
307  {
308  using (var fileStream = File.Create(Path.Combine(executionDir, Path.GetFileName(file) ?? file)))
309  {
310  package.PackageReader.GetStream(file).CopyTo(fileStream);
311  }
312  }
313 
314  return files.Select(file => Path.GetFileName(file) ?? file);
315  }
316  finally
317  {
318  package.Dispose();
319  }
320  }
321  }
