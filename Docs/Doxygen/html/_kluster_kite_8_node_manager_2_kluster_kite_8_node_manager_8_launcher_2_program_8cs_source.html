300  private NodeStartUpConfiguration GetConfig()
301  {
302  while (true)
303  {
304  if ((this.CurrentApiToken == null || this.CurrentApiToken.IsExpired) && !this.GetToken())
305  {
306  return this.GetFallBackConfig();
307  }
308 
309  var authenticator = new OAuth2AuthorizationRequestHeaderAuthenticator(
310  this.CurrentApiToken.AccessToken,
311  "Bearer");
312  var client = new RestClient(this.ConfigurationUrl) { Timeout = 5000, Authenticator = authenticator };
313 
314  var request = new RestRequest { Method = Method.POST };
315  Console.WriteLine($"Requesting configuration for {this.ContainerType} with runtime {this.Runtime} and {PackageRepositoryExtensions.CurrentRuntime}");
316  request.AddJsonBody(
317  new NewNodeTemplateRequest
318  {
319  ContainerType = this.ContainerType,
320  NodeUid = this.Uid,
321  FrameworkRuntimeType = PackageRepositoryExtensions.CurrentRuntime,
322  Runtime = this.Runtime
323  });
324 
325  var response = client.ExecuteTaskAsync<NodeStartUpConfiguration>(request).GetAwaiter().GetResult();
326 
327  if (response.ResponseStatus != ResponseStatus.Completed
328  || response.StatusCode == HttpStatusCode.BadGateway)
329  {
330  return this.GetFallBackConfig();
331  }
332 
333  if (response.StatusCode != HttpStatusCode.OK)
334  {
335  throw new Exception($"Unexpected response (response code: {response.StatusCode}) from service");
336  }
337 
338  return response.Data;
339  }

361  private bool GetToken()
362  {
363  this.CurrentApiToken = null;
364  var client = new RestClient(this.AuthenticationUrl) { Timeout = 5000 };
365 
366  var request = new RestRequest { Method = Method.POST };
367  request.AddParameter("grant_type", "client_credentials");
368  request.AddParameter("client_id", this.ApiClientId);
369  request.AddParameter("client_secret", this.ApiClientSecret);
370 
371  while (true)
372  {
373  var result = client.ExecuteTaskAsync(request).GetAwaiter().GetResult();
374  if (result.ResponseStatus != ResponseStatus.Completed
375  || result.StatusCode == HttpStatusCode.BadGateway
376  || result.StatusCode == HttpStatusCode.NotFound)
377  {
378  return false;
379  }
380 
381  if (result.StatusCode == HttpStatusCode.OK)
382  {
383  this.CurrentApiToken = JsonConvert.DeserializeObject<Token>(result.Content);
384  return true;
385  }
386 
387  Console.WriteLine($@"!!!Failed to authenticate. Response status is {result.StatusCode}. Retrying in 60s");
388  Thread.Sleep(TimeSpan.FromSeconds(60));
389  }

478  private void Start()
479  {
480  switch (this.StopMode)
481  {
482  case EnStopMode.CleanRestart:
483  while (true)
484  {
485  // this is infinity cycle. LaunchNode will exit on node stop. This happens on manual or automatic system shutdown or major exception. In both cases node will be reinitiated.
486  try
487  {
488  this.ConfigureNode();
489  this.LaunchNode();
490  Thread.Sleep(TimeSpan.FromSeconds(30));
491  }
492 #if APPDOMAIN
493  catch (ThreadAbortException)
494  {
495  throw;
496  }
497 #endif
498  catch (Exception exception)
499  {
500  Console.WriteLine($@"Failed to launch, {exception.Message}");
501  Console.WriteLine(exception.StackTrace);
502  Thread.Sleep(TimeSpan.FromSeconds(10));
503  }
504  }
505

507  Console.WriteLine(@"Unsupported stop mode");
