54  /// <summary>
55  /// Gets or sets a value indicating whether the resulting privilege test will be run against combined string of specified privilege and action method name.
56  /// </summary>
57  /// <example>
58  /// For example specified privilege is "KlusterKite.Object" and controller action called is "Get", then user will require privilege "KlusterKite.Object.Get" in order to call this method
59  /// </example>
60  public bool CombinePrivilegeWithActionName { get; set; }
61 
62  /// <inheritdoc />
63  public void OnAuthorization(AuthorizationFilterContext context)
64  {
65  var controllerContext = context.ActionDescriptor as ControllerActionDescriptor;
66  var requiredPrivilege = this.CombinePrivilegeWithActionName
67  ? $"{this.privilege}.{controllerContext?.ActionName}"
68  : this.privilege;
69 
70  var session = context.HttpContext.GetSession();
71  var isAuthorized = session == null
72  || (session.User == null && this.IgnoreOnClientOwnBehalf)
73  || session.UserScope.Contains(requiredPrivilege);
74 
75  if (!isAuthorized)
76  {
77  SecurityLog.CreateRecord(
78  EnSecurityLogType.OperationDenied,
79  this.Severity,
80  context.HttpContext.GetRequestDescription(),
81  "Attempt to access {ControllerName} action {ActionName} without required user privilege {Privilege}",
82  controllerContext?.ControllerName,
83  controllerContext?.ActionName,
84  this.privilege);
85  context.Result = new StatusCodeResult(401);
86  }
87  }
88  }
