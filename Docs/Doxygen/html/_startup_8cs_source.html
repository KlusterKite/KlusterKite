177  private IEnumerable<IWebHostingConfigurator> GetConfigurators(Config config)
178  {
179  var configSection = config.GetConfig("KlusterKite.Web.Configurators");
180  if (configSection == null)
181  {
182  yield break;
183  }
184 
185  foreach (var valuePair in configSection.AsEnumerable())
186  {
187  if (!valuePair.Value.IsString())
188  {
189  continue;
190  }
191 
192  var typeName = valuePair.Value.GetString();
193  if (!string.IsNullOrWhiteSpace(typeName))
194  {
195  var type = Type.GetType(typeName);
196  if (type != null)
197  {
198  var configurator = Activator.CreateInstance(type) as IWebHostingConfigurator;
199  if (configurator != null)
200  {
201  yield return configurator;
202  }
203  }
204  }
205  }
