158  /// <summary>
159  /// Sets the resource current position
160  /// </summary>
161  private void SetPosition()
162  {
163  if (string.IsNullOrWhiteSpace(this.CurrentPoint))
164  {
165  this.Position = EnResourcePosition.NotCreated;
166  }
167  else if (string.IsNullOrWhiteSpace(this.DestinationPoint))
168  {
169  this.Position = EnResourcePosition.Obsolete;
170  }
171  else if (this.SourcePoint == this.CurrentPoint && this.DestinationPoint == this.CurrentPoint)
172  {
173  this.Position = EnResourcePosition.SourceAndDestination;
174  }
175  else if (this.SourcePoint == this.CurrentPoint)
176  {
177  this.Position = EnResourcePosition.Source;
178  }
179  else if (this.DestinationPoint == this.CurrentPoint)
180  {
181  this.Position = EnResourcePosition.Destination;
182  }
183  else if (this.MigrationToDestinationExecutor != null || this.MigrationToSourceExecutor != null)
184  {
185  this.Position = EnResourcePosition.InScope;
186  }
187  else
188  {
189  this.Position = EnResourcePosition.OutOfScope;
