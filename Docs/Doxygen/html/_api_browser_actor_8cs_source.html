113  private void OnApiEndpointDown(Address address)
114  {
115  bool needsSchemaRebuild = false;
116  Context.GetLogger()
117  .Info("{Type}: API endpoint node {NodeAddress} was removed", this.GetType().Name, address.ToString());
118 
119  List<string> apiNames;
120  if (this.knownActiveNodes.TryGetValue(address, out apiNames))
121  {
122  foreach (var apiName in apiNames)
123  {
124  ApiEndpoints endPoint;
125  if (!this.apiEndpoints.TryGetValue(apiName, out endPoint))
126  {
127  Context.GetLogger().Error(
128  "{Type}: on removing {NodeAddress} api {ApiName} provider encounterwed an error: api was not registered",
129  this.GetType().Name,
130  address.ToString(),
131  apiName);
132  continue;
133  }
134 
135  IActorRef handler;
136  endPoint.Provider.Endpoints.TryRemove(address, out handler);
137 
138  Version version;
139  if (!endPoint.HandlerDescriptions.TryGetValue(address, out version))
140  {
141  Context.GetLogger().Error(
142  "{Type}: on removing {NodeAddress} api {ApiName} provider encounterwed an error: api version was not registered",
143  this.GetType().Name,
144  address.ToString(),
145  apiName);
146  continue;
147  }
148 
149  endPoint.HandlerDescriptions.Remove(address);
150  if (endPoint.HandlerDescriptions.Values.All(v => v != version))
151  {
152  endPoint.Descriptions.Remove(version);
153  }
154 
155  if (!endPoint.Descriptions.Any())
156  {
157  needsSchemaRebuild = true;
158  this.apiEndpoints.Remove(apiName);
159  Context.GetLogger()
160  .Info("{Type}: API {ApiName} is now unsupported", this.GetType().Name, apiName);
161  }
162  else if (endPoint.Descriptions.Keys.Max() < version)
163  {
164  needsSchemaRebuild = true;
165  }
166  }
167  }
168 
169  this.knownActiveNodes.Remove(address);
170  this.unknownActiveNodes.Remove(address);
171 
172  if (needsSchemaRebuild)
173  {
174  this.RebuildSchema();
175  }
176  }

255  private void RebuildSchema()
256  {
257  if (!this.apiEndpoints.Any())
258  {
259  this.schemaProvider.CurrentSchema = null;
260  Context.GetLogger().Info("{Type}: There is now api to support", this.GetType().Name);
261  return;
262  }
263 
264  var descriptions =
265  this.apiEndpoints.Values.Select(
266  d =>
267  new
268  {
269  Description = d.Descriptions.Values.OrderByDescending(ad => ad.Version).FirstOrDefault(),
270  d.Provider
271  }).Where(d => d.Description != null).ToList();
272 
273  // renews the provider descriptions. This descriptions are used only in schema generation process
274  foreach (var description in descriptions)
275  {
276  description.Provider.Description = description.Description;
277  }
278 
279  try

299  private void AddHandlers(Address address, List<ApiDiscoverResponse> descriptions)
300  {
301  bool needsSchemaRebuild = false;
302  foreach (var description in descriptions)
303  {
304  ApiEndpoints endpoints;
305  if (!this.apiEndpoints.TryGetValue(description.Description.ApiName, out endpoints))
306  {
307  needsSchemaRebuild = true;
308  endpoints = new ApiEndpoints(
309  description.Description,
310  Context.System);
311  this.apiEndpoints.Add(description.Description.ApiName, endpoints);
312  Context.GetLogger()
313  .Info(
314  "{Type}: API {ApiName} is now supported",
315  this.GetType().Name,
316  description.Description.ApiName);
317  }
318 
319  var currentMaximumApiVersion = endpoints.Descriptions.Any()
320  ? endpoints.Descriptions.Keys.Max()
321  : new Version(0, 0);
322 
323  if (description.Description.Version > currentMaximumApiVersion)
324  {
325  needsSchemaRebuild = true;
326  }
327 
328  if (!endpoints.Descriptions.ContainsKey(description.Description.Version))
329  {
330  endpoints.Descriptions.Add(description.Description.Version, description.Description);
331  }
332 
333  endpoints.Provider.Endpoints.TryAdd(address, description.Handler);
334  endpoints.HandlerDescriptions[address] = description.Description.Version;
335  }
336 
337  if (needsSchemaRebuild)
338  {
339  this.RebuildSchema();
