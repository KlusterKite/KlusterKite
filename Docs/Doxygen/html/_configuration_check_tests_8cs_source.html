51  /// <summary>

53  /// </summary>
54  /// <returns>
55  /// The async task
56  /// </returns>
57  [Fact]
58  public async Task DefaultConfigurationNoError()
59  {
60  var configuration = CreateConfiguration();
61  var errors =
62  (await configuration.SetPackagesDescriptionsForTemplates(CreateRepository(), new List<string> { Net46, NetCore })).ToList();
63  this.WriteErrors(errors);
64  Assert.Equal(0, errors.Count);
65 
66  var template = configuration.Settings.NodeTemplates.First();
67  Assert.NotNull(template.PackagesToInstall);
68  Assert.True(template.PackagesToInstall.ContainsKey(CurrentRuntime));
69  var packagesToInstall = template.PackagesToInstall[CurrentRuntime];
70  Assert.Equal(
71  "dp1 1.0.0; dp2 1.0.0; p1 1.0.0; p2 1.0.0",
72  string.Join("; ", packagesToInstall.OrderBy(p => p.Id).Select(p => $"{p.Id} {p.Version}")));
73  }
