62  {
63  var schemaProvider = this.Container.Resolve<SchemaProvider>();
64  Assert.Null(schemaProvider.CurrentSchema);

66  this.ExpectNoMsg();
67  Assert.Null(schemaProvider.CurrentSchema);
68 
69  var nodeAddress = new UniqueAddress(new Address("akka.tcp", "KlusterKite", "testNode1", 1), 1);
70  var remoteActorRef =
71  new FakeActorRef(
72  ActorPath.Parse($"{nodeAddress.Address}/user/KlusterKite/API/Publisher"),
73  this.TestActor);
74 
75  browser.Tell(
76  new ClusterEvent.MemberUp(
77  ClusterExtensions.MemberCreate(nodeAddress, 1, MemberStatus.Up, ImmutableHashSet<string>.Empty)));
78 
79  // node role check
80  this.ExpectNoMsg();
81  browser.Tell(
82  new ClusterEvent.MemberUp(
83  ClusterExtensions.MemberCreate(
84  nodeAddress,
85  1,
86  MemberStatus.Up,
87  ImmutableHashSet.Create("KlusterKite.API.Endpoint"))));
88 
89  var discoverRequest = this.ExpectMsg<RemoteTestMessage<ApiDiscoverRequest>>();
90  Assert.Equal("/user/KlusterKite/API/Publisher", discoverRequest.ReceiverPath);
91  Assert.Equal(nodeAddress.Address, discoverRequest.RecipientAddress);
92  this.ExpectNoMsg();
93  Assert.Null(schemaProvider.CurrentSchema);
94 
95  this.TestScheduler.Advance(TimeSpan.FromSeconds(30));
96  discoverRequest = this.ExpectMsg<RemoteTestMessage<ApiDiscoverRequest>>();
97  Assert.Equal("/user/KlusterKite/API/Publisher", discoverRequest.ReceiverPath);
98  Assert.Equal(nodeAddress.Address, discoverRequest.RecipientAddress);
99  this.ExpectNoMsg();
100  Assert.Null(schemaProvider.CurrentSchema);
101 
102  var internalApiProvider = new API.Tests.Mock.TestProvider();
103  var discoverResponse = new ApiDiscoverResponse
104  {
105  Description = internalApiProvider.ApiDescription,
106  Handler = this.TestActor
107  };
108 
109  browser.Tell(new List<ApiDiscoverResponse> { discoverResponse }, remoteActorRef);
110  this.ExpectNoMsg();
111 
112  // The new api node now launched
113  Assert.NotNull(schemaProvider.CurrentSchema);
114  this.TestScheduler.Advance(TimeSpan.FromSeconds(30));
115  this.ExpectNoMsg();
116 
117  browser.Tell(
118  new ClusterEvent.MemberRemoved(
119  ClusterExtensions.MemberCreate(
120  nodeAddress,
121  1,
122  MemberStatus.Removed,
123  ImmutableHashSet.Create("KlusterKite.API.Endpoint")),

126  this.ExpectNoMsg();
