173  private static async Task<IEnumerable<PackageIdentity>> GetPackagesToInstall(
174  IPackageRepository repository,
175  SeederConfiguration seederConfiguration)
176  {
177  // TODO: extract to helper method along with Configuration extensions
178  var supportedFramework = NuGetFramework.ParseFrameworkName(
179  PackageRepositoryExtensions.CurrentRuntime,
180  DefaultFrameworkNameProvider.Instance);
181 
182  var packageTasks =
183  seederConfiguration.RequiredPackages.Select(
184  async packageName =>
185  {
186  var package = await repository.GetAsync(packageName);
187  if (package == null)
188  {
189  throw new PackageNotFoundException(packageName);
190  }
191 
192  return package;
193  });
194 
195  var packagesToInstall = (await Task.WhenAll(packageTasks)).ToDictionary(p => p.Identity.Id);
196  var queue = new Queue<IPackageSearchMetadata>(packagesToInstall.Values);
197 
198  while (queue.Count > 0)
199  {
200  var package = queue.Dequeue();
201  var dependencySet =
202  NuGetFrameworkUtility.GetNearest(package.DependencySets, supportedFramework);
203  if (dependencySet == null || !NuGetFrameworkUtility.IsCompatibleWithFallbackCheck(
204  supportedFramework,
205  dependencySet.TargetFramework))
206  {
207  continue;
208  }
209 
210  foreach (var dependency in dependencySet.Packages)
211  {
212  IPackageSearchMetadata packageToInstall;
213  if (!packagesToInstall.TryGetValue(dependency.Id, out packageToInstall))
214  {
215  packageToInstall = await repository.GetAsync(dependency.Id);
216  if (packageToInstall == null)
217  {
218  throw new PackageNotFoundException(dependency.Id);
219  }
220 
221  packagesToInstall.Add(dependency.Id, packageToInstall);
222  if (queue.All(p => p.Identity.Id != packageToInstall.Identity.Id))
223  {
224  queue.Enqueue(packageToInstall);
225  }
226  }
227  }
228  }
229 
230  return packagesToInstall.Values.Select(p => p.Identity);
231  }

238  private static Configuration ReadConfiguration(string[] args)
239  {
240  var configuration = new Configuration();
241  configuration.ConfigFile = args != null && args.Length > 0 ? args[0] : "seeder.hocon";
242  if (configuration.ConfigFile == null || !File.Exists(configuration.ConfigFile))
243  {
244  Console.WriteLine("Configuration file was not found");
245  return null;
246  }
247 
248  configuration.Config = ConfigurationFactory.ParseString(File.ReadAllText(configuration.ConfigFile));
249  configuration.Nuget = configuration.Config.GetString("KlusterKite.NodeManager.PackageRepository");
250  if (string.IsNullOrWhiteSpace(configuration.Nuget))
251  {
252  Console.WriteLine("Nuget is not properly defined");
253  return null;
254  }
255 
256  configuration.NugetCheckPeriod =
257  configuration.Config.GetTimeSpan("NugetCheckPeriod", TimeSpan.FromMinutes(1));
258  configuration.Runtime = configuration.Config.GetString("Runtime");
259 
260  configuration.Configurations = new List<SeederConfiguration>();
261  foreach (var subConfig in configuration.Config.GetStringList("SeederConfigurations"))
262  {
263  var seederConfig = configuration.Config.GetConfig(subConfig);
264  if (seederConfig == null)
265  {
266  Console.WriteLine($"SeederConfiguration {subConfig} was not found in the config");
267  return null;
268  }
269 
270  var seederConfiguration = new SeederConfiguration(subConfig, seederConfig);
271  if (seederConfiguration.RequiredPackages == null || seederConfiguration.RequiredPackages.Count == 0)
272  {
273  Console.WriteLine($"SeederConfiguration {subConfig} misses the RequiredPackages property");
274  return null;
275  }
276 
277  if (seederConfiguration.Seeders == null || seederConfiguration.Seeders.Count == 0)
278  {
279  Console.WriteLine($"SeederConfiguration {subConfig} misses the Seeders property");
280  return null;
281  }
282 
283  configuration.Configurations.Add(seederConfiguration);
284  }
285 
286  if (configuration.Configurations.Count == 0)
287  {
288  Console.WriteLine("SeederConfiguration is empty");
289  return null;
290  }
291 
292  return configuration;
293  }
294  }
