102  /// <summary>
103  /// Process the <see cref="ConfigurationSetReadyRequest"/>
104  /// </summary>
105  /// <param name="request">
106  /// The request
107  /// </param>
108  /// <returns>
109  /// The <see cref="Task"/>.
110  /// </returns>
111  private async Task OnConfigurationCheck(ConfigurationCheckRequest request)
112  {
113  try
114  {
115  using (var ds = this.GetContext())
116  {
117  Context.GetLogger().Info("{Type}: checking configuration {ConfigurationId}", this.GetType().Name, request.Id);
118  var configuration = ds.Configurations.FirstOrDefault(r => r.Id == request.Id);
119  if (configuration == null)
120  {
121  this.Sender.Tell(CrudActionResponse<Configuration>.Error(new EntityNotFoundException(), null));
122  Context.GetLogger().Info(
123  "{Type}: checking configuration {ConfigurationId} - not found",
124  this.GetType().Name,
125  request.Id);
126  return;
127  }
128 
129  if (configuration.State != EnConfigurationState.Draft)
130  {
131  this.Sender.Tell(
132  CrudActionResponse<Configuration>.Error(
133  new Exception("Only draft configurations can be checked"),
134  null));
135  Context.GetLogger().Info(
136  "{Type}: checking configuration {ConfigurationId} - not draft",
137  this.GetType().Name,
138  request.Id);
139  return;
140  }
141 
142  var supportedFrameworks =
143  Context.System.Settings.Config.GetStringList("KlusterKite.NodeManager.SupportedFrameworks");
144 
145  Context.GetLogger().Info(
146  "{Type}: checking configuration {ConfigurationId} against frameworks {Frameworks}",
147  this.GetType().Name,
148  request.Id,
149  string.Join(", ", supportedFrameworks));
150  var errors = await configuration.CheckAll(ds, this.nugetRepository, supportedFrameworks.ToList());
151  if (errors.Count > 0)
152  {
153  this.Sender.Tell(
154  CrudActionResponse<Configuration>.Error(new MutationException(errors.ToArray()), null));
155  Context.GetLogger().Info(
156  "{Type}: checking configuration {ConfigurationId} completed",
157  this.GetType().Name,
158  request.Id);
159  return;
160  }
161 
162  this.Sender.Tell(CrudActionResponse<Configuration>.Success(configuration, null));
163  }
164  }
165  catch (Exception exception)
166  {
167  this.Sender.Tell(CrudActionResponse<Configuration>.Error(exception, null));
168  }
169  }
