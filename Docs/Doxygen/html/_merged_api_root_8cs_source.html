159  /// <summary>
160  /// Creates an api requests to gather all data
161  /// </summary>
162  /// <param name="context">
163  /// The request contexts
164  /// </param>
165  /// <param name="requestContext">
166  /// The request Context.
167  /// </param>
168  /// <returns>
169  /// The request data
170  /// </returns>
171  private async Task<JObject> DoApiRequests(ResolveFieldContext context, RequestContext requestContext)
172  {
173  var taskList = new List<Task<JObject>>();
174  foreach (var provider in this.Providers.Select(fp => fp.Provider))
175  {
176  var request = this.GatherMultipleApiRequest(provider, context.FieldAst, context).ToList();
177  if (request.Count > 0)
178  {
179  taskList.Add(provider.GetData(request, requestContext));
180  }
181  }
182 
183  JObject data;
184  if (taskList.Count == 0)
185  {
186  data = new JObject();
187  }
188  else
189  {
190  var responses = await Task.WhenAll(taskList);
191  var options = new JsonMergeSettings
192  {
193  MergeArrayHandling = MergeArrayHandling.Merge,
194  MergeNullValueHandling = MergeNullValueHandling.Ignore
195  };
196 
197  var response = responses.Aggregate(
198  new JObject(),
199  (seed, next) =>
200  {
201  seed.Merge(next, options);
202  return seed;
203  });
204  data = this.ResolveData(context, response);
205  }
206 
207  if (data.Property(GlobalIdPropertyName) != null)
208  {
209  data.Property(GlobalIdPropertyName).Value = new JArray();
210  }
211  else
212  {
213  data.Add(GlobalIdPropertyName, new JArray());
214  }
215 
216  return data;
217  }

354  }
355 
356  var edgeType = responseType.EdgeType;
357  var nodeType = responseType.EdgeType.ObjectType;
358  var requestedFields = new List<ApiRequest>();
359  var idSubRequestRequest = new List<ApiRequest>
360  {
361  new ApiRequest
362  {
363  FieldName =
364  nodeType.KeyField.FieldName,
365  Alias = "__id"
366  }
367  };

371  FieldName = "result",
372  Fields = idSubRequestRequest
373  };

376  var topFields =
377  GetRequestedFields(context.FieldAst.SelectionSet, context, this.mergedField.Type).ToList();
378 
379  var nodeRequests = topFields.Where(f => f.Name == "node" || f.Name == "edge").ToList();
380 
381  foreach (var nodeRequest in nodeRequests)
382  {
383  var nodeAlias = nodeRequest.Alias ?? nodeRequest.Name;
384  switch (nodeRequest.Name)
385  {
386  case "node":
387  var nodeFields = nodeType.GatherSingleApiRequest(nodeRequest, context).ToList();
388  nodeFields.Add(new ApiRequest { Alias = "__id", FieldName = nodeType.KeyField.FieldName });
389  requestedFields.Add(
390  new ApiRequest { Alias = nodeAlias, FieldName = "result", Fields = nodeFields });
391  break;
392  case "edge":
393  var edgeFields = new List<ApiRequest>();
394  foreach (var edgeNodeRequests in
395  GetRequestedFields(nodeRequest.SelectionSet, context, edgeType)
396  .Where(f => f.Name == "node"))
397  {
398  edgeFields.AddRange(

400  f =>

404  return f;
405  }));
406  }
407 
408  edgeFields.Add(new ApiRequest { Alias = "__id", FieldName = nodeType.KeyField.FieldName });
409  requestedFields.Add(
410  new ApiRequest { Alias = nodeAlias, FieldName = "result", Fields = edgeFields });
411 
412  break;
413  }
414  }
415 
416  if (responseType.ErrorType != null)
417  {
418  var errorsRequest = topFields.Where(f => f.Name == "errors");
419  foreach (var field in errorsRequest)
420  {
421  requestedFields.Add(
422  new ApiRequest
423  {
424  FieldName = "errors",
425  Alias = field.Alias,
426  Fields =

428  .ToList()
429  });
430  }
431  }

434  {
435  Arguments = arguments,
436  FieldName = this.mergedField.FieldName,
437  Fields = requestedFields
438  };
439

441  if (data != null)
442  {
443  var mutation = (ApiMutation)this.mergedField.OriginalFields[this.provider.Description.ApiName];
444  var treePath = mutation.Path.Take(mutation.Path.Count - 1).ToList();
445 
446  var parentGlobalId = new JArray(treePath.Select(r => new JObject { { "f", r.FieldName } }));
447  data.Add(GlobalIdPropertyName, parentGlobalId);
448 
449  var elementRequest = mutation.Path.LastOrDefault();
450  if (elementRequest != null)
451  {
452  var localRequest = new JObject { { "f", elementRequest.FieldName } };
453  data.Add(RequestPropertyName, localRequest);
454  }
455  }
