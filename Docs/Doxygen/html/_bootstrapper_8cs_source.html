115  /// <summary>
116  /// Creates configuration file from environment variables, start-up parameters and local files
117  /// </summary>
118  /// <param name="configurations">Start up parameters</param>
119  /// <returns>Top level configuration</returns>
120  /// <remarks>
121  /// Load priority:
122  /// * List of files in start up parameters in order of parameters
123  /// * environment variables
124  /// * akka.hocon file in local application directory
125  /// * Akka config section in application configuration file
126  /// </remarks>
127  private static Config CreateTopLevelConfig(string[] configurations)
128  {
129  var config = ConfigurationFactory.Empty;
130  if (configurations != null)
131  {
132  Console.WriteLine($@"loading config from command line parameters: {string.Join(", ", configurations)}");
133  config = configurations
134  .Where(File.Exists)
135  .Select(File.ReadAllText)
136  .Aggregate(
137  config,
138  (current, configText) => current.WithFallback(ConfigurationFactory.ParseString(configText)));
139  }
140 
141  Console.WriteLine(@"loading config from environment");
142  var networkName = Environment.GetEnvironmentVariable("NETWORK_NAME");
143  if (!string.IsNullOrEmpty(networkName))
144  {
145  config = config.WithFallback(ConfigurationFactory.ParseString($"{{ akka.remote.helios.tcp.public-hostname = \"{networkName.Replace("\"", "\\\"")}\" }}"));
146  }
147 
148  // ReSharper disable once AssignNullToNotNullAttribute
149  var hoconPath = Path.Combine(Path.GetDirectoryName(typeof(Bootstrapper).GetTypeInfo().Assembly.Location), "akka.hocon");
150  Console.WriteLine($@"loading akka.hocon ({hoconPath})");
151  if (File.Exists(hoconPath))
152  {
153  Console.WriteLine($@"reading file ({hoconPath})");
154  var hoconConfig = File.ReadAllText(hoconPath);
155  Console.WriteLine(@"file loaded in memory");
156  config = config.WithFallback(ConfigurationFactory.ParseString(hoconConfig));
157  }
158  else
159  {
160  Log.Warning("File {fileName} was not found", hoconPath);
161  }
162 
163  Console.WriteLine(@"loading application configuration");
164 
165  return config;
166  }
