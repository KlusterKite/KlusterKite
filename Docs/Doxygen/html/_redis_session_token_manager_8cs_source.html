65  this.system = system;
66  var connectionString = system.Settings.Config.GetString("KlusterKite.Security.SessionRedis.RedisConnection");
67 
68  var config = ConfigurationOptions.Parse(connectionString);
69  var addressEndpoint = (DnsEndPoint)config.EndPoints.First();
70  var port = addressEndpoint.Port;
71 
72  bool isIp = IsIpAddress(addressEndpoint.Host);
73  if (!isIp)
74  {
75  var ip = Dns.GetHostEntryAsync(addressEndpoint.Host).GetAwaiter().GetResult();
76  config.EndPoints.Remove(addressEndpoint);
77  config.EndPoints.Add(ip.AddressList.First(), port);
78  }
79 
80  this.redisConnectionString = config;
81 
82  this.redisDb = system.Settings.Config.GetInt("KlusterKite.Security.SessionRedis.RedisDb");

86  /// <inheritdoc />
87  public Task<string> CreateAccessToken(AccessTicket session)
88  {
89  var tokenUid = Guid.NewGuid();
90  var token = tokenUid.ToString("N");
91 
92  var data = session.SerializeToAkka(this.system);
93  using (var connection = ConnectionMultiplexer.Connect(this.redisConnectionString))
94  {
95  var db = connection.GetDatabase(this.redisDb);
96 
97  var result = db.StringSet(
98  this.GetRedisAccessKey(token),
99  data,
100  session.Expiring.HasValue ? (TimeSpan?)(session.Expiring.Value - DateTimeOffset.Now) : null);
101 
102  if (result)
103  {
104  return Task.FromResult(token);
105  }
106 
107  var exception = new Exception("Session data server is unavailable");
108  this.system.Log.Error(exception, "{Type}: Session data server is unavailable", this.GetType().Name);
109  throw exception;
110  }
111  }
