54  /// <summary>

56  /// </summary>
57  /// <param name="request">The request</param>

59  {
60  var system = Context.System;
61  this.apiProvider.ResolveMutation(
62  request,
63  request.Context,
64  exception => system.Log.Error(exception, "{Type}: mutation resolve exception", this.GetType().Name))
65  .PipeTo(
