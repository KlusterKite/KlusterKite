135  public static IQueryable<TObject> ApplySorting(
136  IQueryable<TObject> query,
137  IEnumerable<SortingCondition> sortingConditions)
138  {
139  if (sortingConditions == null || query == null)
140  {
141  return query;
142  }
143 
144  var sorted =
145  sortingConditions.Where(c => Properties.ContainsKey(c.PropertyName))
146  .Select(c => new { c.Direction, Property = Properties[c.PropertyName] })
147  .ToList();
148 
149  if (sorted.Count == 0)
150  {
151  return query;
152  }
153 
154  var firstCondition = sorted[0];
155  var type = typeof(TObject);
156  var parameter = Expression.Parameter(type, "e");
157  var property = Expression.Property(parameter, firstCondition.Property);
158  var expression = Expression.Lambda(property, parameter);
159 
160  var sortedQuery = firstCondition.Direction == SortingCondition.EnDirection.Asc
161  ? (IOrderedQueryable<TObject>)OrderByMethods[firstCondition.Property.PropertyType].Invoke(null, new object[] { query, expression })
162  : (IOrderedQueryable<TObject>)OrderByDescendingMethods[firstCondition.Property.PropertyType].Invoke(null, new object[] { query, expression });
163 
164  foreach (var sort in sorted.Skip(1))
165  {
166  parameter = Expression.Parameter(type, "e");
167  property = Expression.Property(parameter, sort.Property);
168  expression = Expression.Lambda(property, parameter);
169 
170  sortedQuery = sort.Direction == SortingCondition.EnDirection.Asc
171  ? (IOrderedQueryable<TObject>)ThenByMethods[sort.Property.PropertyType].Invoke(null, new object[] { sortedQuery, expression })
172  : (IOrderedQueryable<TObject>)ThenByDescendingMethods[sort.Property.PropertyType].Invoke(null, new object[] { sortedQuery, expression });
173  }
