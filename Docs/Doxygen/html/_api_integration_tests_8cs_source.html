191  /// <summary>
192  /// Authenticate as admin user
193  /// </summary>
194  /// <param name="client">The rest client</param>
195  /// <returns>The access token</returns>
196  private static async Task<string> Authenticate(RestClient client)
197  {
198  var authenticationRequest = new RestRequest { Method = Method.POST, Resource = "/api/1.x/security/token" };
199  authenticationRequest.AddParameter("grant_type", "password");
200  authenticationRequest.AddParameter("username", "admin");
201  authenticationRequest.AddParameter("password", "admin");
202  authenticationRequest.AddParameter("client_id", WebApplication.WebApplicationClientId);
203  var authenticationResult = await client.ExecuteTaskAsync<TokenResponse>(authenticationRequest);
204 
205  Assert.Equal(ResponseStatus.Completed, authenticationResult.ResponseStatus);
206  Assert.Equal(HttpStatusCode.OK, authenticationResult.StatusCode);
207  Assert.NotNull(authenticationResult.Data);
208  Assert.NotNull(authenticationResult.Data.AccessToken);
209  return authenticationResult.Data.AccessToken;
210  }
