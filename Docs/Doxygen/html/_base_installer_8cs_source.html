88  /// </param>

90  /// Akka and system configuration
91  /// </returns>
92  public static Config GetStackedConfig(ContainerBuilder container, Config config)
93  {
94  Serilog.Log.Information("KlusterKite starting plugin manager");
95 
96  List<BaseInstaller> list;
97  if (!RegisteredInstallers.TryGetValue(container, out list))
98  {
99  return config;
100  }
101 
102  var rolesList = list.SelectMany(i => i.GetRoles()).Distinct().Where(r => !string.IsNullOrWhiteSpace(r))
103  .Select(r => $"\"{r.Replace("\"", "\\\"")}\"").ToList();
104 
105  if (rolesList.Any())
106  {
107  config = config.WithFallback(
108  ConfigurationFactory.ParseString($"akka.cluster.roles = [{string.Join(", ", rolesList)}]"));
109  }
110 
111  return list.OrderByDescending(i => i.AkkaConfigLoadPriority).Aggregate(
112  config,
113  (current, installer) => current.WithFallback(installer.GetAkkaConfig()));
114  }

171  {
172  if (builder == null)
173  {
174  throw new ArgumentNullException(nameof(builder));
175  }
176 
177  if (config == null)
178  {
179  throw new ArgumentNullException(nameof(config));
180  }
181 
182  List<BaseInstaller> installers;
183  if (!RegisteredInstallers.TryGetValue(builder, out installers))
184  {
185  throw new InvalidOperationException("builder is not registered");
186  }
187 
188  foreach (var installer in installers)
189  {
190  installer.RegisterComponents(builder, config);
191  builder.RegisterInstance(installer).As<BaseInstaller>();
192  }

217  /// <summary>

219  /// </summary>
220  /// <param name="container">The builder.</param>
221  public void Install([NotNull] ContainerBuilder container)
222  {
223  if (container == null)
224  {
225  throw new ArgumentNullException(nameof(container));
226  }
227 
228  lock (RegisteredInstallers)

230  if (!RegisteredInstallers.ContainsKey(container))

233  }
234 
235  if (RegisteredInstallers[container].Contains(this))
236  {
237  return;

239 
240  RegisteredInstallers[container].Add(this);
