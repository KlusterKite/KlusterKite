113  ApiRequest request,
114  RequestContext context,
115  Action<Exception> onErrorCallback)
116  {
117  try
118  {
119  ObjectResolver.MutationDescription mutation;
120 
121  if (!this.mutations.TryGetValue(request.FieldName, out mutation))
122  {
123  return null;
124  }
125 
126  var resolveResult = await mutation.ResolveContainer(this, context, this.argumentsSerializer, onErrorCallback);
127  if (resolveResult == null)
128  {
129  return null;
130  }
131 
132  request.FieldName = mutation.Field.Name;
133  var rootRequest = new ApiRequest { Fields = new List<ApiRequest> { request } };
134 
135  var objectResolver = mutation.Resolver as ObjectResolver;
136  if (objectResolver != null)
137  {
138  var result = await objectResolver.ResolveQuery(
139  resolveResult,
140  rootRequest,
141  mutation.Field,
142  context,
143  this.argumentsSerializer,
144  onErrorCallback) as JObject;
145 
146  return mutation.Type == ApiMutation.EnType.Connection
147  ? result?.Property(request.FieldName)?.Value as JObject
148  : new JObject { { "result", result?.Property(request.FieldName)?.Value } };
149  }
150 
151  var connectionResolver = mutation.Resolver as IConnectionResolver;
152  if (connectionResolver != null)
153  {
154  var resolvedMutation = await connectionResolver.ResolveMutation(
155  resolveResult,
156  request,
157  mutation.Field,
158  context,
159  this.argumentsSerializer,
160  onErrorCallback);
161 
162  return resolvedMutation;
163  }
164 
165  return null;
166  }
167  catch (Exception e)
168  {
169  onErrorCallback?.Invoke(e);
170  return null;
171  }
172  }

177  private void Assemble()
178  {
179  this.ApiDescription.Version = this.GetType().GetTypeInfo().Assembly.GetName().Version;
180  this.ApiDescription.Types.Clear();
181  this.ApiDescription.Mutations.Clear();
182 
183  var rootResolver = (ObjectResolver)Activator.CreateInstance(typeof(ObjectResolver<>)
184  .MakeGenericType(this.GetType()));
185  this.resolver = rootResolver;
186  var root = rootResolver.GetApiType();
187  this.ApiDescription.TypeName = this.ApiDescription.ApiName = root.TypeName;
188  this.ApiDescription.Description = root.Description;
189  this.ApiDescription.Fields = new List<ApiField>(root.Fields.Select(f => f.Clone()));
190 
191  List<ApiType> allTypes;

193  List<string> errors;
194  rootResolver.CreateApiRoot(out allTypes, out this.argumentsSerializer, out mutationList, out errors);
195  this.generationErrors.Clear();
196  this.generationErrors.AddRange(errors);
197  this.mutations = mutationList.ToDictionary(m => m.MutationName);
198  this.ApiDescription.Types = allTypes;
199  this.ApiDescription.Mutations = this.mutations.Values.Select(d => d.CreateMutationField()).ToList();
200  }
