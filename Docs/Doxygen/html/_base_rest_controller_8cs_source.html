270  protected virtual async Task<TObject> SendRequest(CrudActionMessage<TObject, TId> request)
271  {
272  CrudActionResponse<TObject> result;
273  if (DataIsLarge)
274  {
275  var notification = request.ActionType == EnActionType.Get
276  ? await this.System.ActorSelection(this.GetDbActorProxyPath())
277  .Ask<ParcelNotification>(request, this.AkkaTimeout)
278  : await this.System.GetParcelManager()
279  .Ask<ParcelNotification>(
280  new Parcel
281  {
282  Payload = request,
283  Recipient =
284  this.System.ActorSelection(
285  this.GetDbActorProxyPath())
286  },
287  this.AkkaTimeout);
288 
289  result = (CrudActionResponse<TObject>)await notification.Receive(this.System);
290  }
291  else
292  {
293  result =
294  await this.System.ActorSelection(this.GetDbActorProxyPath())
295  .Ask<CrudActionResponse<TObject>>(request, this.AkkaTimeout);
296  }
297 
298  if (result.Exception != null && result.Exception.GetType() != typeof(EntityNotFoundException))
299  {
300  throw result.Exception;
301  }
302 
303  if (result.Exception?.GetType() == typeof(EntityNotFoundException) || result.Data == null)
304  {
305  return null;
306  }
307 
308  return result.Data;
309  }
