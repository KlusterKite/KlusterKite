130  private void InitHostFromConfiguration(string hostName, Config config)
131  {
132  StringBuilder hostConfig = new StringBuilder();
133  foreach (var parameter in config.AsEnumerable())
134  {
135  if (parameter.Value.IsString())
136  {
137  hostConfig.AppendFormat("\t{0} {1};\n", parameter.Key, parameter.Value.GetString());
138  }
139 
140  if (parameter.Value.IsObject()

142  {

144  this.InitServiceFromConfiguration(

146  serviceName,

148  }
149  }

160  private void InitServiceFromConfiguration(HostConfiguration host, string serviceName, Config config)

181  if (headers == null

183  && (headers.GetString() ?? string.Empty).ToLower()
184  .IndexOf("Host ", StringComparison.OrdinalIgnoreCase) < 0)
185  || (headers.IsArray()
186  && headers.GetArray()
187  .Select(v => v.GetString())
188  .All(
189  v =>

191  {
192  serviceConfig.Append("\t\tproxy_set_header Host $http_host;\n");
193  }

202  private void OnNewNodeDescription(WebDescriptionResponse description)
203  {
204  var nodeAddress = this.Sender.Path.Address;
205  if (nodeAddress.Host == null)
206  {
207  // supposed this is local address
208  nodeAddress = Cluster.Get(Context.System).SelfAddress;
209  }
210 
211  if (!this.KnownActiveNodes.Contains(nodeAddress))
212  {
213  // node managed to go down before it was initialized
214  return;
215  }
216 
217  if (this.NodePublishUrls.ContainsKey(nodeAddress))
218  {
219  // duplicate configuration info
220  return;
221  }
222 
223  this.NodePublishUrls[nodeAddress] = description;
224 
225  foreach (var serviceDescription in description.Services)
226  {
227  var nodeServiceDescription = new NodeServiceConfiguration
228  {
229  NodeAddress = nodeAddress,
230  ServiceDescription = serviceDescription
231  };
232  var serviceConfiguration = this.Configuration[serviceDescription.PublicHostName][serviceDescription.Route];
233  if (!serviceConfiguration.ActiveNodes.Contains(nodeServiceDescription))
234  {
235  serviceConfiguration.ActiveNodes.Add(nodeServiceDescription);
236  }
237  }
238 
239  this.WriteConfiguration();
240  }

246  private void OnWebNodeDown(Address nodeAddress)
247  {
248  this.KnownActiveNodes.Remove(nodeAddress);
249  WebDescriptionResponse description;
250  if (!this.NodePublishUrls.TryGetValue(nodeAddress, out description))
251  {
252  // something strange. Local data is corrupted;
253  return;
254  }
255 
256  foreach (var serviceHost in description.Services.GroupBy(s => s.PublicHostName))
257  {
258  var host = this.Configuration[serviceHost.Key];
259  foreach (var serviceDescription in serviceHost)
260  {
261  var activeNodes = host[serviceDescription.Route].ActiveNodes;
262  var d = activeNodes.FirstOrDefault(n => n.NodeAddress == nodeAddress);
263  if (d != null)
264  {
265  activeNodes.Remove(d);
266  }
267  }
268 
269  host.Flush();
270  }
271 
272  this.Configuration.Flush();
273  this.NodePublishUrls.Remove(nodeAddress);
274  this.WriteConfiguration();
275  }
