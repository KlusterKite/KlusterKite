110  /// <summary>

112  /// </summary>
113  /// <param name="obj">The original message</param>
114  protected virtual void ForwardMessage(object obj)
115  {
116  var messageToSendType = typeof(TestMessage<>).MakeGenericType(obj.GetType());
117  var messageToSend = Activator.CreateInstance(messageToSendType);
118  messageToSendType.GetProperty("Message")?.SetValue(messageToSend, obj);
