190  /// Process collection requests
191  /// </summary>
192  /// <typeparam name="TObject">

197  /// </typeparam>
198  /// <param name="collectionRequest">Collection request</param>
199  /// <returns>The list of objects</returns>
200  protected virtual async Task OnCollectionRequest<TObject, TId>(CollectionRequest<TObject> collectionRequest)
201  where TObject : class
202  {
203  try
204  {
205  using (var ds = this.GetContext())
206  {
207  int maxLimit;
208  if (!this.classQueryLimits.TryGetValue(
209  NamingUtilities.ToCSharpRepresentation(typeof(TObject)),
210  out maxLimit))
211  {
212  maxLimit = this.defaultQueryLimit;
213  }
214 
215  var limit = maxLimit != 0 && (!collectionRequest.Count.HasValue
216  || collectionRequest.Count.Value > maxLimit)
217  ? maxLimit
218  : collectionRequest.Count;
219 
220  var factory = DataFactory<TContext, TObject, TId>.CreateFactory(this.ComponentContext, ds);
221 
222  var response = await factory.GetList(
223  collectionRequest.Filter,
224  collectionRequest.Sort,
225  collectionRequest.Skip,
226  limit,
227  collectionRequest.ApiRequest);
228 
229  if (collectionRequest.AcceptAsParcel)
230  {
231  Context.GetParcelManager().Tell(
232  new Parcel { Payload = response, Recipient = this.Sender },
233  this.Self);
234  }
235  else
236  {
237  this.Sender.Tell(response);
238  }
239  }
240  }
241  catch (Exception exception)
242  {
243  try
244  {
245  Context.GetLogger().Error(
246  exception,
247  "{Type}: Exception on processing CollectionRequest\n\t filter: {FilterExpression}\n\t sort: {SortExpression}\n\t limit: {Limit}\n\t offset: {Offset}",
248  $"BaseCrudActor<{typeof(TObject).Name}>",
249  collectionRequest.Filter?.ToString(),
250  collectionRequest.Sort?.ToString(),
251  collectionRequest.Count,
252  collectionRequest.Skip);
253  }
254  catch
255  {

258  "{Type}: Exception on processing CollectionRequest",
259  $"BaseCrudActor<{typeof(TObject).Name}>");
260  }
261 
262  var response = new CollectionResponse<TObject> { Items = new List<TObject>() };
263  if (collectionRequest.AcceptAsParcel)
264  {

348  /// <summary>
349  /// Request process method
350  /// </summary>
351  /// <typeparam name="TObject">
352  /// The type of ef object
353  /// </typeparam>
354  /// <typeparam name="TId">
355  /// The type of object identity field
356  /// </typeparam>
357  /// <param name="request">
358  /// The action request
359  /// </param>
360  /// <returns>
361  /// Execution task
362  /// </returns>
363  [UsedImplicitly]
364  protected virtual async Task<CrudActionResponse<TObject>> ProcessRequest<TObject, TId>(
365  CrudActionMessage<TObject, TId> request)
366  where TObject : class
367  {
368  using (var ds = this.GetContext())
369  {
370  var factory = DataFactory<TContext, TObject, TId>.CreateFactory(this.ComponentContext, ds);
371  switch (request.ActionType)
372  {
373  case EnActionType.Get:
374  try
375  {
376  var result = await factory.Get(request.Id);
377  if (result.HasValue)
378  {
379  result = this.OnSelect(result.Value);
380  }
381 
382  // security read log should be set on client endpoint
383  return result.HasValue
384  ? CrudActionResponse<TObject>.Success(result, request.ExtraData)
385  : CrudActionResponse<TObject>.Error(
386  new EntityNotFoundException(),
387  request.ExtraData);
388  }
389  catch (Exception exception)
390  {
391  return CrudActionResponse<TObject>.Error(
392  new DatasourceInnerException("Exception on Get operation", exception),
393  request.ExtraData);
394  }
395 
396  case EnActionType.Create:
397  {
398  var entity = request.Data;
399  if (entity == null)
400  {
401  return CrudActionResponse<TObject>.Error(
402  new RequestEmptyException(),
403  request.ExtraData);
404  }
405 
406  entity = this.BeforeCreate(entity);
407  var oldObject = await factory.Get(factory.GetId(entity));
408  if (oldObject != null)
409  {
410  var crudActionResponse = CrudActionResponse<TObject>.Error(
411  new InsertDuplicateIdException(),
412  request.ExtraData);
413  crudActionResponse.Data = oldObject;
414 
415  return crudActionResponse;
416  }
417 
418  if (entity == null)
419  {
420  return CrudActionResponse<TObject>.Error(
421  new BeforeActionException(),
422  request.ExtraData);
423  }
424 
425  try
426  {
427  await factory.Insert(entity);
428 
429  // security update logs are set here to be sure that they are made independently of client notification success
430  SecurityLog.CreateRecord(
431  EnSecurityLogType.DataCreateGranted,
432  entity is ICrucialObject ? EnSeverity.Crucial : EnSeverity.Trivial,
433  request.RequestContext,
434  "{ObjectType} with {ObjectId} id was created",
435  typeof(TObject).FullName,
436  factory.GetId(entity));
437 
438  this.AfterCreate(entity);
439  return CrudActionResponse<TObject>.Success(entity, request.ExtraData);
440  }
441  catch (Exception exception)
442  {
443  return CrudActionResponse<TObject>.Error(
444  new DatasourceInnerException("Exception on Insert operation", exception),
445  request.ExtraData);
446  }
447  }
448 
449  case EnActionType.Update:
450  {
451  var entity = request.Data;
452  if (entity == null)
453  {
454  return CrudActionResponse<TObject>.Error(
455  new RequestEmptyException(),
456  request.ExtraData);
457  }
458 
459  var oldObject = await factory.Get(request.Id);
460  if (oldObject == null)
461  {
462  return CrudActionResponse<TObject>.Error(
463  new EntityNotFoundException(),
464  request.ExtraData);
465  }
466 
467  if (request.ApiRequest != null)
468  {
469  var updatedObject = await factory.Get(request.Id);
470  if (updatedObject == null)
471  {
472  return CrudActionResponse<TObject>.Error(
473  new EntityNotFoundException(),
474  request.ExtraData);
475  }
476 
477  DataUpdater<TObject>.Update(updatedObject, entity, request.ApiRequest);
478  entity = updatedObject;
479  }
480 
481  entity = this.BeforeUpdate<TObject>(entity, oldObject);
482  if (entity == null)
483  {
484  return CrudActionResponse<TObject>.Error(
485  new BeforeActionException(),
486  request.ExtraData);
487  }
488 
489  try
490  {
491  await factory.Update(entity, oldObject);
492 
493  // security update logs are set here to be sure that they are made independently of client notification success
494  if (!factory.GetId(entity).Equals(factory.GetId(oldObject)))
495  {
496  SecurityLog.CreateRecord(
497  EnSecurityLogType.DataUpdateGranted,
498  entity is ICrucialObject ? EnSeverity.Crucial : EnSeverity.Trivial,
499  request.RequestContext,
500  "{ObjectType} with id {ObjectId} was updated. New id is {NewObjectId}",
501  typeof(TObject).FullName,
502  factory.GetId(oldObject),
503  factory.GetId(entity));
504  }
505  else
506  {
507  SecurityLog.CreateRecord(
508  EnSecurityLogType.DataUpdateGranted,
509  entity is ICrucialObject ? EnSeverity.Crucial : EnSeverity.Trivial,
510  request.RequestContext,
511  "{ObjectType} with id {ObjectId} was updated.",
512  typeof(TObject).FullName,
513  factory.GetId(entity));
514  }
515 
516  this.AfterUpdate<TObject>(entity, oldObject);
517  return CrudActionResponse<TObject>.Success(entity, request.ExtraData);
518  }
519  catch (Exception exception)
520  {
521  return CrudActionResponse<TObject>.Error(
522  new DatasourceInnerException("Exception on Update operation", exception),
523  request.ExtraData);
524  }
525  }
526 
527  case EnActionType.Delete:
528  {
529  var oldObject = await factory.Get(request.Id);
530  if (oldObject == null)
531  {
532  return CrudActionResponse<TObject>.Error(
533  new EntityNotFoundException(),
534  request.ExtraData);
535  }
536 
537  oldObject = this.BeforeDelete(oldObject.Value);
538  if (oldObject == null)
539  {
540  return CrudActionResponse<TObject>.Error(
541  new BeforeActionException(),
542  request.ExtraData);
543  }
544 
545  try
546  {
547  oldObject = await factory.Delete(factory.GetId(oldObject));
548  }
549  catch (Exception exception)
550  {
551  return CrudActionResponse<TObject>.Error(
552  new DatasourceInnerException("Exception on Delete operation", exception),
553  request.ExtraData);
554  }
555 
556  if (oldObject == null)
557  {
558  return CrudActionResponse<TObject>.Error(
559  new EntityNotFoundException("After \"Before\" action modification"),
560  request.ExtraData);
561  }
562 
563  // security update logs are set here to be sure that they are made independently of client notification success
564  SecurityLog.CreateRecord(
565  EnSecurityLogType.DataDeleteGranted,
566  oldObject.Value is ICrucialObject ? EnSeverity.Crucial : EnSeverity.Trivial,
567  request.RequestContext,
568  "{ObjectType} with id {ObjectId} was deleted.",
569  typeof(TObject).FullName,
570  factory.GetId(oldObject));
571 
572  this.AfterDelete<TObject>(oldObject);
573  return CrudActionResponse<TObject>.Success(oldObject, request.ExtraData);
574  }
575 
576  default:
577  return CrudActionResponse<TObject>.Error(
578  new ArgumentOutOfRangeException(nameof(request.ActionType)),
579  request.ExtraData);
580  }
581  }
582  }
