125  {

127  {

130 
131  if (!this.Resources.Contains(resourceId.ConnectionString))
132  {
133  throw new InvalidOperationException($"Unknown connection {resourceId.ConnectionString}");
134  }
135 
136  if (!this.DefinedMigrationPoints.Contains(pointToMigrate))
137  {
138  throw new InvalidOperationException($"Unknown migration point {pointToMigrate} among {string.Join(", ", this.DefinedMigrationPoints)}");
139  }
140 
141  var path = Path.Combine(this.Config.GetString("TestMigrator.Dir"), resourceId.ConnectionString);
142  SetMigrationPoint(path, pointToMigrate);
143  yield return "success";
