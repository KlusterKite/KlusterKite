39  public static IEnumerable<string> CheckSchema(Schema schema)
40  {
41  var types = schema.AllTypes.ToDictionary(t => t.Name);
42  var checkedInputGraphTypes = new List<string>();
43  foreach (var graphType in
44  types.Values.OfType<IComplexGraphType>()
45  .Where(g => !g.GetType().FullName.StartsWith("GraphQL.Introspection")))
46  {
47  foreach (var field in graphType.Fields)
48  {
49  var resolvedType = GetResolvedType(field.ResolvedType);
50  if (resolvedType?.Name == null || !types.ContainsKey(resolvedType.Name))
51  {
52  yield return $"Field {field.Name} of type {graphType.Name} has unregistered type";
53  }
54 
55  if (field.Arguments == null)
56  {
57  continue;
58  }
59 
60  foreach (var argument in field.Arguments)
61  {
62  resolvedType = GetResolvedType(argument.ResolvedType);
63  IGraphType argumentType;
64  if (resolvedType?.Name == null || !types.TryGetValue(resolvedType.Name, out argumentType))
65  {
66  yield return
67  $"Field {field.Name} of type {graphType.Name} "
68  + $"has argument {argument.Name} unregistered type";
69  continue;
70  }
71 
72  if (!(argumentType is ScalarGraphType) && !(argumentType is IInputGraphType)
73  && !(argumentType is InputObjectGraphType))
74  {
75  yield return
76  $"Field {field.Name} of type {graphType.Name} "
77  + $"has argument {argument.Name} has invalid type {argumentType.Name}";
78  continue;
79  }
80 
81  var complexGraphType = argumentType as IComplexGraphType;
82  if (complexGraphType == null)
83  {
84  continue;
85  }
86 
87  foreach (var error in CheckInputGraphType(complexGraphType, checkedInputGraphTypes))
88  {
89  yield return error;
90  }
91  }
92  }
93 
94  if (!graphType.Fields.Any())
95  {
96  yield return $"Type {graphType.Name} has no fields";
97  }
98  }
99  }

106  public static IEnumerable<string> CheckSchemaIntrospection(Schema schema)
107  {
108  var result = new DocumentExecuter().ExecuteAsync(
109  r =>
110  {
111  r.Schema = schema;
112  r.Query = Queries.IntrospectionQuery;
113  r.UserContext = new RequestContext();
114  }).Result;
115  var response = new DocumentWriter(true).Write(result);
116  var json = JObject.Parse(response);
117  var types =
118  (json.SelectToken("data.__schema.types") as JArray)?.ToDictionary(
119  p => ((JObject)p).Property("name")?.Value,
120  p => (JObject)p);
121  if (types == null)
122  {
123  yield return "Could not get types list via introspection";
124  yield break;
125  }
126 
127  var inputTypesChecked = new List<string>();
128  foreach (var type in types.Values.Where(t => !t.Property("name").Value.ToObject<string>().StartsWith("__")))
129  {
130  var typeName = type.Property("name").Value.ToObject<string>();
131  var fields = type.Property("fields")?.Value as JArray;
132  if (fields != null)
133  {
134  foreach (var field in fields)
135  {
136  var fieldType = field.SelectToken("type.kind")?.ToObject<string>() == "LIST"
137  ? field.SelectToken("type.ofType.name")?.ToObject<string>()
138  : field.SelectToken("type.name")?.ToObject<string>();
139  var fieldName = field.SelectToken("name")?.ToObject<string>();
140  if (!types.ContainsKey(fieldType ?? string.Empty))
141  {
142  yield return $"{typeName} property {fieldName} has unknown type {fieldType}";
143  }
144 
145  var arguments = field.SelectToken("args") as JArray;
146  if (arguments == null || arguments.Count == 0)
147  {
148  continue;
149  }
150 
151  foreach (var argument in arguments)
152  {
153  var argumentType = argument.SelectToken("type.kind")?.ToObject<string>() == "LIST"
154  ? argument.SelectToken("type.ofType.name")?.ToObject<string>()
155  : argument.SelectToken("type.name")?.ToObject<string>();
156  var argumentName = argument.SelectToken("name")?.ToObject<string>();
157 
158  JObject argumentTypeJson;
159  if (!types.TryGetValue(argumentType ?? string.Empty, out argumentTypeJson))
160  {
161  yield return
162  $"{typeName} property {fieldName} "
163  + $"has argument {argumentName} of unknown type {argumentType}";
164  }
165  else
166  {
167  var argumentTypeKind = argumentTypeJson.SelectToken("kind").ToObject<string>();
168  if (argumentTypeKind != "SCALAR" && argumentTypeKind != "ENUM"
169  && argumentTypeKind != "INPUT_OBJECT")
170  {
171  yield return
172  $"{typeName} property {fieldName} "
173  + $"has argument {argumentName} of type {argumentType} "
174  + $"of unsupported kind {argumentTypeKind}";
175  }
176  else if (argumentTypeKind == "INPUT_OBJECT")
177  {
178  foreach (
179  var error in
180  CheckIntrospectionInputType(argumentTypeJson, types, inputTypesChecked))
181  {
182  yield return error;
183  }
184  }
185  }
186  }

197  public static Schema Generate(List<ApiProvider> providers)
198  {
199  var createdTypes = new Dictionary<string, MergedType>();
200  var api = MergeApis(providers, createdTypes);
201  api.Initialize();
202  var nodeInterface = new NodeInterface();
203  var root = new MergedRoot("Query", providers, api);
204  root.Initialize();
205  createdTypes[root.ComplexTypeName] = root;
206  createdTypes[api.ComplexTypeName] = api;
207 
208  var types = createdTypes.Values.GroupBy(t => t.ComplexTypeName).Select(g => g.First()).ToList();
209 
210  var allInterfaces = GenerateApiTypeInterfaces(providers, types, nodeInterface);
211 
212  IObjectGraphType mutationType;
213  var graphTypes = GenerateApiTypes(providers, types, nodeInterface, allInterfaces, api, out mutationType);
214  AdjustInterfaceFieldsArguments(allInterfaces, graphTypes);
215 
216  graphTypes.Values.OfType<VirtualGraphType>().ForEach(vgt => vgt.StoreFieldResolvers());
217 
218  var schema = new Schema
219  {
220  Query = (VirtualGraphType)graphTypes[root.ComplexTypeName],
221  Mutation = mutationType.Fields.Any() ? mutationType : null
222  };
223 
224  var arrayOfInterfaces = allInterfaces.Values.OfType<TypeInterface>().Cast<IGraphType>().ToArray();
225  schema.RegisterTypes(arrayOfInterfaces);
226  schema.Initialize();
227 
228  return schema;
229  }

328  private static IEnumerable<string> CheckIntrospectionInputType(
329  JObject type,
330  IReadOnlyDictionary<JToken, JObject> types,
331  ICollection<string> typesChecked)
332  {
333  var kind = type.SelectToken("kind").ToObject<string>();
334  var typeName = type.Property("name").Value.ToObject<string>();
335  if (typesChecked.Contains(typeName))
336  {
337  yield break;
338  }
339 
340  typesChecked.Add(typeName);
341 
342  if (kind != "SCALAR" && kind != "ENUM" && kind != "INPUT_OBJECT")
343  {
344  yield return $"{typeName} is not a valid input type";
345  }
346 
347  if (kind != "INPUT_OBJECT")
348  {
349  yield break;
350  }
351 
352  var fields = type.Property("inputFields")?.Value as JArray;
353  if (fields == null || fields.Count == 0)
354  {
355  yield return $"{typeName} has no fields";
356  yield break;
357  }
358 
359  foreach (var field in fields)
360  {
361  var fieldTypeName = field.SelectToken("type.kind")?.ToObject<string>() == "LIST"
362  ? field.SelectToken("type.ofType.name")?.ToObject<string>()
363  : field.SelectToken("type.name")?.ToObject<string>();
364  var fieldName = field.SelectToken("name")?.ToObject<string>();
365  JObject fieldType;
366  if (!types.TryGetValue(fieldTypeName ?? string.Empty, out fieldType))
367  {
368  yield return $"{typeName} property {fieldName} has unknown type {fieldTypeName}";
369  continue;
370  }
371 
372  var fieldKind = fieldType.SelectToken("kind").ToObject<string>();
373  if (fieldKind != "SCALAR" && fieldKind != "ENUM" && fieldKind != "INPUT_OBJECT")
374  {
375  yield return
376  $"Input type {typeName} property {fieldName} "
377  + $"of type {fieldTypeName} of unsupported kind {fieldKind}";
378  }
379  else if (fieldKind == "INPUT_OBJECT")
380  {
381  foreach (var error in CheckIntrospectionInputType(fieldType, types, typesChecked))
382  {
383  yield return error;
384  }
385  }
386  }

466  ApiProvider provider,

469  ICollection<string> path,
470  bool createAsInput,
471  Dictionary<string, MergedType> typesCreated)
472  {
473  MergedType createdType;
474  if (apiField.ScalarType != EnScalarType.None)
475  {
476  return CreateScalarType(apiField.ScalarType, typesCreated);
477  }
478 
479  var apiType = provider.Description.Types.FirstOrDefault(t => t.TypeName == apiField.TypeName);
480  if (apiType == null)
481  {
482  throw new Exception("type was not found");
483  }
484 
485  var apiEnumType = apiType as ApiEnumType;
486  if (apiEnumType != null)
487  {
488  return CreateEnumType(apiEnumType, provider, typesCreated);
489  }

492  if (apiField.Flags.HasFlag(EnFieldFlags.IsConnection))

495  apiField.Arguments.Where(a => a.Flags.HasFlag(EnFieldFlags.IsTypeArgument))

499  }

509  }
510

512  {
513  return createdType;
514  }
515 
516  typesCreated[objectType.ComplexTypeName] = objectType;
517 
518  MergeFields(
519  objectType,
520  apiObjectType.Fields,
521  provider,

794  private static void MergeFields(
795  MergedObjectType parentType,
796  IEnumerable<ApiField> apiFields,
797  ApiProvider provider,
798  ICollection<string> path,
799  bool createAsInput,
800  Dictionary<string, MergedType> typesCreated)
801  {
802  var fields = createAsInput
803  ? apiFields.Where(
804  f =>
805  f.Flags.HasFlag(EnFieldFlags.CanBeUsedInInput)
806  && f.Arguments.All(a => a.Flags.HasFlag(EnFieldFlags.IsTypeArgument)))
807  : apiFields.Where(f => f.Flags.HasFlag(EnFieldFlags.Queryable));
808 
809  foreach (var apiField in fields)
810  {
811  MergedField complexField;
812  if (parentType.Fields.TryGetValue(apiField.Name, out complexField))
813  {
814  if (apiField.ScalarType != EnScalarType.None || createAsInput
815  || apiField.Flags.HasFlag(EnFieldFlags.IsConnection)
816  || apiField.Flags.HasFlag(EnFieldFlags.IsArray) || !(complexField.Type is MergedObjectType)
817  || complexField.Arguments.Any() || apiField.Arguments.Any())
818  {
819  // todo: write merge error
820  continue;
821  }
822  }
823 
824  var flags = apiField.Flags;
825 
826  if (createAsInput && flags.HasFlag(EnFieldFlags.IsConnection))
827  {
828  flags &= ~EnFieldFlags.IsConnection;
829  flags |= EnFieldFlags.IsArray;
830  }
831 
832  var cloneField = apiField.Clone();
833  cloneField.Flags = flags;
834 
835  var fieldType = CreateMergedType(provider, cloneField, complexField, path, createAsInput, typesCreated);
836 
837  if (fieldType == null)
838  {
839  continue;
840  }
841 
842  var fieldArguments = new Dictionary<string, MergedField>();
843 
844  if (!createAsInput)
845  {
846  foreach (var argument in apiField.Arguments)
847  {
848  var fieldArgumentType = CreateMergedType(provider, argument, null, path, true, typesCreated);
849  fieldArguments[argument.Name] = new MergedField(
850  argument.Name,
851  fieldArgumentType,
852  provider,
853  apiField,
854  argument.Flags,
855  description: argument.Description);
856  }
857  }
858 
859  var description = string.Join(
860  "\n",
861  new[] { complexField?.Description, apiField.Description }.Where(s => !string.IsNullOrWhiteSpace(s)));
862 
863  var field = new MergedField(
864  apiField.Name,
865  fieldType,
866  provider,
867  cloneField,
868  flags,
869  fieldArguments,
870  string.IsNullOrWhiteSpace(description) ? null : description);
871  if (complexField != null)
872  {
873  foreach (var complexFieldProvider in complexField.Providers)
874  {
875  field.AddProvider(
876  complexFieldProvider,
877  complexField.OriginalFields[complexFieldProvider.Description.ApiName]);
878  }
879  }
880 
881  parentType.Fields[apiField.Name] = field;
882  }
