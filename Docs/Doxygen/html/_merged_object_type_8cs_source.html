118  /// <summary>
119  /// Gets the globalId from source data and object id value
120  /// </summary>
121  /// <param name="source">The source data</param>
122  /// <param name="id">The id value</param>
123  /// <returns>The global id</returns>
124  public static JArray GetGlobalId(JObject source, JToken id)
125  {
126  var parent = source.Parent;
127  JArray parentGlobalId = null;
128  var selfRequest = source.Property(RequestPropertyName)?.Value as JObject;
129  while (parent != null)
130  {
131  parentGlobalId = (parent as JObject)?.Property(GlobalIdPropertyName)?.Value as JArray;
132  if (selfRequest == null)
133  {
134  selfRequest = (parent as JObject)?.Property(RequestPropertyName)?.Value as JObject;
135  }
136 
137  if (parentGlobalId != null)
138  {
139  break;
140  }
141 
142  parent = parent.Parent;
143  }
144 
145  if (source.Property(GlobalIdPropertyName) == null && selfRequest != null)
146  {
147  var globalId = parentGlobalId != null ? new JArray(parentGlobalId) : new JArray();
148  var selfPart = (JObject)selfRequest.DeepClone();
149  if (id != null)
150  {
151  selfPart.Add("id", id);
152  }
153 
154  globalId.Add(selfPart);
155  return globalId;
156  }
157 
158  return null;
159  }

277  requestedFields.Join(

279  s => s.Name,

281  (s, fp) => new { Ast = s, Field = fp.Value }).ToList();
282 
283  foreach (var usedField in usedFields)
284  {
285  var apiField = usedField.Field.OriginalFields[provider.Description.ApiName];
286  if (apiField != null
287  && !apiField.CheckAuthorization(context.UserContext as RequestContext, EnConnectionAction.Query))

291  : EnSeverity.Trivial;

294  EnSecurityLogType.OperationDenied,
295  severity,
296  context.UserContext as RequestContext,

366  {

373  context.Arguments

375  .ToDictionary(p => p.Key, p => p.Value);
376  if (args.Any())

378  var argumentsValue = JObject.FromObject(args);
379  localRequest.Add("a", argumentsValue);

381  }

384  }
