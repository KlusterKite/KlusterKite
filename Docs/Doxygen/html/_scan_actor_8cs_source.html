141  /// <summary>
142  /// Processes the scan request
143  /// </summary>
144  private void OnScanRequest()
145  {
146  Node result;
147 
148  if (this.lastScanResult == null || (DateTimeOffset.Now - this.lastScanTime) > this.scanMemoize)
149  {
150  var cluster = Cluster.Get(Context.System);
151  Context.System.Log.Info("{Type}: scan started", this.GetType().Name);
152  result = this.GetActorMap();
153  result.Name = this.nodeTemplateName != null
154  ? $"{this.nodeTemplateName} {cluster.SelfAddress.Host}:{cluster.SelfAddress.Port}"
155  : $"{cluster.SelfAddress.Host}:{cluster.SelfAddress.Port}";
156  this.lastScanTime = DateTimeOffset.Now;
157  this.lastScanResult = result;
158  Context.System.Log.Info("{Type}: scan finished", this.GetType().Name);
159  }
160  else
161  {
162  result = this.lastScanResult;
163  Context.System.Log.Info("{Type}: Sending scan cache", this.GetType().Name);
164  }
165 
166  Context.GetParcelManager().Tell(new Parcel { Payload = result, Recipient = this.Sender }, this.Self);
167  }

169  /// <summary>

172  /// <param name="node">The sub-network</param>
173  private void SetNodeStats(Node node)
174  {
175  if (node.Children != null)
176  {
177  foreach (var child in node.Children)

179  this.SetNodeStats(child);

184  {

186  node.QueueSizeSum = node.QueueSize + node.Children.Sum(c => c.QueueSizeSum);
187  }
188  else
189  {
190  node.MaxQueueSize = 0;
191  node.QueueSizeSum = node.QueueSize;
192  }
193  }
