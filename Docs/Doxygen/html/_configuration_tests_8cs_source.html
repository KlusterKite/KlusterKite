62  var template1 = new NodeTemplate
63  {
64  Code = "template1",
65  Configuration = "1",
66  PackageRequirements = this.GetPackageRequirements("p1; p2")
67  };
68 
69  var template2 = new NodeTemplate
70  {
71  Code = "template2",
72  Configuration = "2",
73  PackageRequirements = this.GetPackageRequirements("p2; p3")
74  };
75 
76  var configuration = new Configuration
77  {
78  MinorVersion = 1,
79  Name = "1",
80  State = EnConfigurationState.Active,
81  Settings =
82  new ConfigurationSettings
83  {
84  NodeTemplates =
85  this.GetList(template1, template2),
86  Packages = this.CreatePackages(
87  "p1;0.0.1",
88  "p2;0.0.1",
89  "p3;0.0.1")
90  }
91  };
92 
93  using (var context = this.CreateContext())
94  {
95  // facepalm - https://github.com/aspnet/EntityFramework/issues/6872
96  var ids = context.Configurations.Select(r => r.Id).OrderByDescending(id => id).ToList();
97 
98  var compatibleTemplates = configuration.GetCompatibleTemplates(context)
99  .OrderByDescending(t => t.CompatibleConfigurationId).ThenBy(o => o.TemplateCode).ToList();
100  Assert.Equal(2, compatibleTemplates.Count);
101  Assert.Equal("template1", compatibleTemplates.Select(t => t.TemplateCode).First());
102 
103  this.output.WriteLine($"Database configuration ids: {string.Join(", ", ids)}");
104  this.output.WriteLine(
105  $"Compatible templates configuration ids: {string.Join(", ", compatibleTemplates.Select(t => t.CompatibleConfigurationId))}");
106 
107  Assert.Equal(ids[0], compatibleTemplates.Select(t => t.CompatibleConfigurationId).First());
108  Assert.Equal(ids[1], compatibleTemplates.Select(t => t.CompatibleConfigurationId).Skip(1).First());
109  }
