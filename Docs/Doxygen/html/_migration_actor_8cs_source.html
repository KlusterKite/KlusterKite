159  /// <summary>
160  /// Creates the resource migration plan
161  /// </summary>
162  /// <param name="request">The list of migrating resources</param>
163  /// <param name="errors">The list of processing errors</param>
164  /// <returns>The migration plan</returns>
165  private MigrationPlan CreateMigrationPlan(List<ResourceUpgrade> request, out List<MigrationLogRecord> errors)
166  {
167  var plan = new MigrationPlan();
168  errors = new List<MigrationLogRecord>();
169  var errorId = -1;
170  foreach (var resourceUpgrade in request)
171  {
172  var error = new MigrationLogRecord
173  {
174  Id = errorId--,
175  Type = EnMigrationLogRecordType.Error,
176  Started = DateTimeOffset.Now,
177  ConfigurationId = this.StateData.Migration.ToConfigurationId,
178  MigratorTemplateCode = resourceUpgrade.TemplateCode,
179  MigratorTypeName = resourceUpgrade.MigratorTypeName,
180  ResourceCode = resourceUpgrade.ResourceCode
181  };
182 
183  var template =
184  this.StateData.MigrationState?.TemplateStates.FirstOrDefault(
185  t => t.Code == resourceUpgrade.TemplateCode);
186  if (template == null)
187  {
188  error.Message = "Migrator template was not found";
189  errors.Add(error);
190  continue;
191  }
192 
193  error.MigratorTemplateName = template.DestinationTemplate?.Name ?? template.SourceTemplate?.Name;
194 
195  var migrator = template.Migrators.FirstOrDefault(m => m.TypeName == resourceUpgrade.MigratorTypeName);
196  if (migrator == null)
197  {
198  error.Message = "Migrator was not found";
199  errors.Add(error);
200  continue;
201  }
202 
203  error.MigratorName = migrator.Name;
204 
205  var resource = migrator.Resources.FirstOrDefault(r => r.Code == resourceUpgrade.ResourceCode);
206  if (resource == null)
207  {
208  error.Message = "Resource was not found";
209  errors.Add(error);
210  continue;
211  }
212 
213  error.ResourceName = resource.Name;
214 
215  var executionSide = resourceUpgrade.Target == EnMigrationSide.Destination
216  ? resource.MigrationToDestinationExecutor
217  : resource.MigrationToSourceExecutor;
218 
219  if (!executionSide.HasValue)
220  {
221  error.Message = "Resource can not be migrated";
222  errors.Add(error);
223  continue;
224  }
225 
226  var sideExecutionPlans = executionSide.Value == EnMigrationSide.Source
227  ? plan.SourceExecution
228  : plan.DestinationExecution;
229 
230  MigratorTemplatePlan templatePlan;
231  if (!sideExecutionPlans.TryGetValue(resourceUpgrade.TemplateCode, out templatePlan))
232  {
233  templatePlan = new MigratorTemplatePlan();
234  templatePlan.Template = executionSide.Value == EnMigrationSide.Source
235  ? template.SourceTemplate
236  : template.DestinationTemplate;
237  sideExecutionPlans[resourceUpgrade.TemplateCode] = templatePlan;
238  }
239 
240  MigratorMigrationsCommand migratorMigrationsCommand;
241  if (!templatePlan.MigratorPlans.TryGetValue(
242  resourceUpgrade.MigratorTypeName,
243  out migratorMigrationsCommand))
244  {
245  migratorMigrationsCommand =
246  new MigratorMigrationsCommand { TypeName = resourceUpgrade.MigratorTypeName };
247  templatePlan.MigratorPlans[resourceUpgrade.MigratorTypeName] = migratorMigrationsCommand;
248  }
249 
250  var migrationPoint = resourceUpgrade.Target == EnMigrationSide.Source
251  ? resource.SourcePoint
252  : resource.DestinationPoint;
253 
254  migratorMigrationsCommand.Resources.Add(resource.Code, migrationPoint);
255  }
256 
257  return plan;
258  }
