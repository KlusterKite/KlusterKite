64

66 
67  Log.Information(
68  "{Type} Standard log: \n\tMinimum level: {MinimumLevel}\n\tIndex format: {IndexFormat}\n\tNodes:\n\t\t{NodeList}\n",
69  this.GetType().FullName,
70  minimumLevel,

72  string.Join("\n\t\t", nodes));
73 
74  var options = new ElasticsearchSinkOptions(nodes.Select(s => new Uri(s)))

78  IndexFormat = indexFormat,
