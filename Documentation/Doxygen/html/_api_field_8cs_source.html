223  public bool CheckAuthorization(RequestContext context, EnConnectionAction action)
224  {
225  var accessTicket = context?.Authentication;
226  if (accessTicket == null)
227  {
228  return !this.RequireAuthenticatedSession;
229  }
230 
231  if (accessTicket.User == null && this.RequireAuthenticatedUserSession)
232  {
233  return false;
234  }
235 
236  if (this.AuthorizationRules == null)
237  {
238  return true;
239  }
240 
241  foreach (var rule in this.AuthorizationRules)
242  {
243  if ((accessTicket.User != null && rule.IgnoreOnUserPresent)
244  || (accessTicket.User == null && rule.IgnoreOnUserNotPresent))
245  {
246  continue;
247  }
248 
249  if (this.Flags.HasFlag(EnFieldFlags.IsConnection))
250  {
251  if (!rule.ConnectionActions.HasFlag(action))
252  {
253  continue;
254  }
255  }
256 
257  var rulePrivilege = rule.Privilege;
258  if (rule.AddActionNameToRequiredPrivilege)
259  {
260  rulePrivilege = $"{rulePrivilege}.{action}";
261  }
262 
263  if (rule.Scope != EnPrivilegeScope.User)
264  {
265  var clientHasPrivilege = accessTicket.ClientScope.Contains(rulePrivilege);
266  if (!clientHasPrivilege
267  && (rule.Scope == EnPrivilegeScope.Both || rule.Scope == EnPrivilegeScope.Client))
268  {
269  return false;
270  }
271 
272  if (clientHasPrivilege
273  && (rule.Scope == EnPrivilegeScope.Any || rule.Scope == EnPrivilegeScope.Client))
274  {
275  continue;
276  }
277  }
278 
279  var userHasPrivilege = accessTicket.UserScope.Contains(rulePrivilege);
280  if (!userHasPrivilege)
281  {
282  return false;
283  }
284  }
285 
286  return true;
287  }
