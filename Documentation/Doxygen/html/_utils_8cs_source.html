117  private static IEnumerable<PrivilegeDescription> SearchDefinedPrivileges(Type container)
118  {
119  return
120  container.GetFields(BindingFlags.Public | BindingFlags.Static | BindingFlags.FlattenHierarchy)
121  .Where(f => f.IsLiteral && !f.IsInitOnly && f.FieldType == typeof(string))
122  .SelectMany(
123  f =>
124  {
125  var descriptionAttribute =
126  (PrivilegeDescriptionAttribute)
127  f.GetCustomAttribute(typeof(PrivilegeDescriptionAttribute));
128  if (descriptionAttribute == null)
129  {
130  return new[]
131  {
132  new PrivilegeDescription(
133  container.GetTypeInfo().Assembly.GetName().Name,
134  null,
135  (string)f.GetValue(null))
136  };
137  }
138 
139  if (descriptionAttribute.Actions != null && descriptionAttribute.Actions.Length > 0)
140  {
141  return
142  descriptionAttribute.Actions.Select(
143  action =>
144  new PrivilegeDescription(
145  container.GetTypeInfo().Assembly.GetName().Name,
146  descriptionAttribute.Description,
147  $"{f.GetValue(null)}.{action}",
148  action,
149  descriptionAttribute.Target));
150  }
151 
152  return new[]
153  {
154  new PrivilegeDescription(
155  container.GetTypeInfo().Assembly.GetName().Name,
156  descriptionAttribute.Description,
157  (string)f.GetValue(null),
158  target: descriptionAttribute.Target)
