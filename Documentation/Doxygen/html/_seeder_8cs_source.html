93  public override void Seed()
94  {
95  var connectionString = this.Config.GetString(ConfigConnectionStringPath);
96  var databaseName = this.Config.GetString(ConfigDatabaseNamePath);
97  var databaseProviderName = this.Config.GetString(ConfigDatabaseProviderNamePath);
98  using (var context =
99  this.contextFactory.CreateContext<ConfigurationContext>(
100  databaseProviderName,
101  connectionString,
102  databaseName))
103  {
104  var databaseCreator = context.GetService<IDatabaseCreator>() as RelationalDatabaseCreator;
105  if (databaseCreator == null)
106  {
107  Console.WriteLine(@"Error - could not check database existence. There is no IDatabaseCreator.");
108  return;
109  }
110 
111  if (databaseCreator.Exists())
112  {
113  Console.WriteLine(@"KlusterKite configuration database is already existing");
114  return;
115  }
116 
117  context.Database.Migrate();
118 
119  this.SetupUsers(context);
120  var configuration =
121  new ConfigurationSettings
122  {
123  NodeTemplates = this.GetNodeTemplates().ToList(),
124  MigratorTemplates = this.GetMigratorTemplates().ToList(),
125  Packages = this.GetPackageDescriptions().GetAwaiter().GetResult(),
126  SeedAddresses = this.GetSeeds().ToList(),
127  NugetFeed = this.Config.GetString("KlusterKite.NodeManager.PackageRepository")
128  };
129 
130  var initialConfiguration = new Configuration
131  {
132  State = EnConfigurationState.Active,
133  Name = "Initial configuration",
134  Started = DateTimeOffset.Now,
135  Settings = configuration
136  };
137 
138  var supportedFrameworks = this.Config.GetStringList("KlusterKite.NodeManager.SupportedFrameworks");
139  var initialErrors =
140  initialConfiguration.SetPackagesDescriptionsForTemplates(this.packageRepository, supportedFrameworks.ToList()).GetAwaiter().GetResult();
141 
142  foreach (var errorDescription in initialErrors)
143  {
144  Console.WriteLine($@"error in {errorDescription.Field} - {errorDescription.Message}");
145  }
146 
147  context.Configurations.Add(initialConfiguration);
148  context.SaveChanges();
149  }
150 
151  Console.WriteLine(@"KlusterKite configuration database created");
152  }
