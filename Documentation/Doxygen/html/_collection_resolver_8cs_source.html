199  var sortArgument = arguments?.Property("sort")?.Value as JArray;

207  if (items == null)

212 
213  SetLog(request, apiField, context, EnConnectionAction.Query);
214 
215  var result = new JObject();
216  var fields = request.Fields.GroupBy(f => f.Alias ?? f.FieldName).Select(
217  g =>
218  {
219  var f = g.First();
220  if (f.Fields == null)
221  {
222  return f;
223  }
224 
225  return new ApiRequest

227  Alias = f.Alias,
228  Arguments = f.Arguments,
229  FieldName = f.FieldName,
230  Fields = g.SelectMany(sr => sr.Fields).ToList()

232  });
233

235  {

237  {

413  private static Expression<Func<T, bool>> CreateFilter(JObject filter, JProperty id)
414  {
415  var filterCheck = filter != null ? CreateFilterPart(filter) : null;
416  var idCheck = id != null ? FilterChecks[NodeMetaData.KeyPropertyName](id) : null;
417 
418  var check = filterCheck != null && idCheck != null
419  ? Expression.And(filterCheck, idCheck)
420  : filterCheck ?? idCheck;
421 
422  return check == null
423  ? null
424  : Expression.Lambda<Func<T, bool>>(check, FilterSourceParameter);
425  }
