179  protected static MutationResult<TObject> CreateResponse(CrudActionResponse<TObject> response)
180  {

183  return new MutationResult<TObject> { Result = response.Data };
184  }
185 
186  var errors = response.Exception.Match<List<ErrorDescription>>()
187  .With<EntityNotFoundException>(
188  e => new List<ErrorDescription> { new ErrorDescription("id", "not found") })
189  .With<MutationException>(e => e.Errors)
190  .ResultOrDefault(
191  e =>
192  {
193  var exception = e as Exception;
194  var errorDescription = new ErrorDescription(
195  "null",
196  exception != null ? $"{exception.Message}\n{exception.StackTrace}" : "unknown error");
197  return new List<ErrorDescription> { errorDescription };
198  });
199 
200  return new MutationResult<TObject> { Errors = errors };
