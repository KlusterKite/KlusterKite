300  private void InitChildActorsFromConfig(IActorContext context, ActorPath actorPath, IComponentContext container)
301  {
302  var config = Context.System.Settings.Config.GetConfig("akka.actor.deployment");
303  if (config == null)
304  {
305  return;
306  }
307 
308  var namespacePathElements = actorPath.Elements.Skip(1).ToList();
309  foreach (var pair in config.AsEnumerable())
310  {
311  var key = pair.Key;
312 
313  var path = key.Split(new[] { '/' }, StringSplitOptions.RemoveEmptyEntries);
314  if (path.Length != namespacePathElements.Count + 1)
315  {
316  continue;
317  }
318 
319  if (namespacePathElements.Where((t, i) => path[i] != t).Any())
320  {
321  continue;
322  }
323 
324  var actorConfig = config.GetConfig(key);
325  EnActorType actorType;
326  if (!Enum.TryParse(actorConfig.GetString("actor-type"), out actorType))
327  {
328  actorType = EnActorType.Simple;
329  }
330 
331  var currentPath = "/" + string.Join("/", namespacePathElements);
332 
333  switch (actorType)
334  {
335  case EnActorType.Singleton:
336  this.CreateSingletonActor(context, actorConfig, currentPath, path.Last());
337  break;
338 
339  case EnActorType.SingletonProxy:
340  this.CreateSingletonProxyActor(context, actorConfig, currentPath, path.Last());
341  break;
342 
343  case EnActorType.Sharding:
344  this.CreateShardingActor(context, actorConfig, container, path.Last());
345  break;
346 
347  case EnActorType.ShardingProxy:
348  this.CreateShardingProxyActor(context, actorConfig, container, path.Last());
349  break;
350 
351  default:
352  this.CreateSimpleActor(context, actorConfig, container, currentPath, path.Last());
353  break;
354  }
355  }
356  }
